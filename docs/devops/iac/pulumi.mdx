# Pulumi：真正的编程式 IaC

「用 Python/TypeScript/Go 写基础设施代码，而不是 YAML。」

Terraform 用 HCL（HashiCorp Configuration Language）描述基础设施，它的语法简洁，但毕竟是 DSL——你不能用 if/else、for/while、函数这些编程语言的基本能力。

Pulumi 走了一条不同的路：**直接用你熟悉的编程语言写基础设施代码**。Python、TypeScript、Go、JavaScript、C#——你的语言能力，就是你的 IaC 能力。

## Pulumi vs Terraform

| 维度 | Terraform | Pulumi |
|---|---|---|
| **配置语言** | HCL (DSL) | Python/TS/Go (编程语言) |
| **逻辑控制** | 有限 | 完整（if/for/函数）|
| **类型系统** | 无 | 强类型（TS/Go）|
| **IDE 支持** | 一般 | 优秀（原生语言）|
| **调试** | plan/apply | print/调试器 |
| **测试** | 有限 | 完整（单元测试）|
| **生态** | 成熟（Provider 多）| 快速增长 |
| **学习曲线** | HCL 特定 | 取决于选择语言 |

## Pulumi 核心概念

### Stack（栈）

Stack 是 Pulumi 的隔离运行环境，类似于 Terraform 的 workspace：

```
┌─────────────────────────────────────────────────────────────┐
│                    Pulumi 项目结构                            │
│                                                              │
│  my-project/                                                │
│  ├── Pulumi.yaml                                            │
│  ├── Pulumi.dev.yaml        ← Dev 栈配置                     │
│  ├── Pulumi.prod.yaml       ← Prod 栈配置                    │
│  └── src/                                                     │
│      ├── index.ts                                           │
│      └── resources/                                          │
│          ├── vpc.ts                                         │
│          └── compute.ts                                      │
└─────────────────────────────────────────────────────────────┘
```

### 程序 vs 栈

```
Pulumi 程序（你的代码）
    │
    ├── 读取栈配置
    ├── 创建资源
    └── 输出结果
    │
    ▼
Pulumi 执行
    │
    ├── 生成执行计划
    ├── 调用云 API
    └── 更新状态
```

```typescript
// 创建标准化的安全组
function createSecurityGroup(
    name: string,
    vpcId: pulumi.Input<string>,
    ingressRules: aws.ec2.SecurityGroupIngress[]
) {
    const sg = new aws.ec2.SecurityGroup(name, {
        vpcId: vpcId,
        ingress: ingressRules,
        egress: [
            {
                protocol: "-1",
                fromPort: 0,
                toPort: 0,
                cidrBlocks: ["0.0.0.0/0"],
            },
        ],
    });
    return sg;
}

// 使用
const webSg = createSecurityGroup("web-sg", vpc.id, [
    { protocol: "tcp", fromPort: 80, toPort: 80, cidrBlocks: ["0.0.0.0/0"] },
    { protocol: "tcp", fromPort: 443, toPort: 443, cidrBlocks: ["0.0.0.0/0"] },
]);
```

### 类封装（OOP）

```typescript
// 定义 VPC 类
class Network {
    vpc: aws.ec2.Vpc;
    subnets: aws.ec2.Subnet[];
    constructor(cidr: string, azs: string[]) {
        this.vpc = new aws.ec2.Vpc("main", {
            cidrBlock: cidr,
            tags: { Name: "main-vpc" },
        });

        this.subnets = azs.map((az, index) =>
            new aws.ec2.Subnet(`subnet-${az}`, {
                vpcId: this.vpc.id,
                cidrBlock: `10.0.${index + 1}.0/24`,
                availabilityZone: az,
            })
        );
    }
}

// 使用
const network = new Network("10.0.0.0/16", ["us-east-1a", "us-east-1b"]);
```

## Python 示例

```python
import pulumi
import pulumi_aws as aws

# 配置
config = pulumi.Config()
environment = config.require("environment")
vpc_cidr = "10.0.0.0/16"

# VPC
vpc = aws.ec2.Vpc(
    "main",
    cidr_block=vpc_cidr,
    enable_dns_hostnames=True,
    enable_dns_support=True,
    tags={
        "Name": f"myapp-{environment}",
        "Environment": environment,
    }
)

# 子网
availability_zones = ["a", "b", "c"]
subnets = []
for i, az in enumerate(availability_zones):
    subnet = aws.ec2.Subnet(
        f"subnet-{az}",
        vpc_id=vpc.id,
        cidr_block=f"10.0.{i+1}.0/24",
        availability_zone=f"us-east-1{az}",
        map_public_ip_on_launch=True,
        tags={
            "Name": f"myapp-{environment}-subnet-{az}",
        }
    )
    subnets.append(subnet)

# 输出
pulumi.export("vpc_id", vpc.id)
pulumi.export("subnet_ids", [s.id for s in subnets])
```

## Go 示例

```go
package main

import (
    "github.com/pulumi/pulumi-aws/sdk/v5/go/aws/ec2"
    "github.com/pulumi/pulumi/sdk/v3/go/pulumi"
)

func main() {
    pulumi.Run(func(ctx *pulumi.Context) error {
        // 创建 VPC
        vpc, err := ec2.NewVpc(ctx, "main", &ec2.VpcArgs{
            CidrBlock:          pulumi.String("10.0.0.0/16"),
            EnableDnsHostnames: pulumi.Bool(true),
            EnableDnsSupport:   pulumi.Bool(true),
            Tags: pulumi.StringMap{
                "Name": pulumi.String("my-vpc"),
            },
        })
        if err != nil {
            return err
        }

        // 创建子网
        subnets := []string{"10.0.1.0/24", "10.0.2.0/24", "10.0.3.0/24"}
        for i, cidr := range subnets {
            _, err := ec2.NewSubnet(ctx, pulumi.Sprintf("subnet-%d", i), &ec2.SubnetArgs{
                VpcId:                  vpc.ID(),
                CidrBlock:              pulumi.String(cidr),
                AvailabilityZone:       pulumi.Stringf("us-east-1%c", 'a'+i),
                MapPublicIpOnLaunch:    pulumi.Bool(true),
            })
            if err != nil {
                return err
            }
        }

        // 导出
        ctx.Export("vpcId", vpc.ID())
        return nil
    })
}
```

```typescript
import * as pulumi from "@pulumi/pulumi";

        // 测试代码...
        const vpc = new aws.ec2.Vpc("test-vpc", {
            cidrBlock: "10.0.0.0/16",
        });

        // 断言
        expect(vpc.cidrBlock).to.equal("10.0.0.0/16");
    });

```

### 集成测试

```typescript
// integration-test.ts
import * as pulumi from "@pulumi/pulumi";

async function test() {
    // 创建资源
    const stack = await pulumi.runTest(async () => {
        // 部署
        const up = await pulumi.up({ stack: "test" });

        // 验证
        const vpcId = up.outputs["vpcId"] as string;
        console.log(`Created VPC: ${vpcId}`);

        // 检查资源
        const ec2 = new aws.sdk.EC2({ region: "us-east-1" });
        const result = await ec2.describeVpcs({ VpcIds: [vpcId] }).promise();
        if (result.Vpcs.length === 0) {
            throw new Error("VPC not found!");
        }

        // 清理
        await pulumi.destroy({ stack: "test" });
    });
}
```

## Pulumi vs Terraform 选型

### 什么时候选 Pulumi？

| 场景 | Pulumi 优势 |
|---|---|
| 复杂的条件逻辑 | 用 if/else 处理不同环境 |
| 动态资源数量 | 用循环根据配置创建资源 |
| 团队熟悉 TypeScript/Python/Go | 复用现有编程技能 |
| 需要单元测试 | 用熟悉的测试框架 |
| 动态配置计算 | 用完整编程能力处理复杂配置 |
| 微服务配置模板 | 用函数/类封装复用逻辑 |

### 什么时候选 Terraform？

| 场景 | Terraform 优势 |
|---|---|
| 团队不熟悉编程语言 | HCL 简单易学 |
| Provider 生态 | Provider 更多更成熟 |
| 社区资源 | 模块和示例更丰富 |
| 状态锁定 | S3+DynamoDB 成熟方案 |
| 简单基础设施 | YAML-like 配置更直观 |

## 总结

Pulumi 代表了 IaC 的一个新范式：

- **真正的代码**：用熟悉的编程语言写基础设施
- **完整能力**：条件、循环、函数、类、测试
- **强类型**：TypeScript/Go 提供类型安全
- **原生 IDE**：代码补全、重构、调试
- **测试友好**：单元测试、集成测试

<Badge text="选型建议" type="tip" />

- **如果你的团队擅长 TypeScript/Python/Go** → Pulumi
- **如果需要复杂的条件逻辑** → Pulumi
- **如果需要完善的测试能力** → Pulumi
- **如果 Provider 生态是首要考虑** → Terraform
- **如果团队不熟悉编程语言** → Terraform

<Badge text="趋势" type="info" /> Pulumi 的编程能力是 IaC 的未来方向。随着 IaC 复杂度增加，Terraform 的 DSL 局限性会越来越明显。
