# Linux 文件系统：Ext4 与 inode

你用 `ls` 查看文件，`cat` 读取内容，`touch` 创建新文件。这些命令是如何工作的？

答案是 **VFS（Virtual File System）**——Linux 的文件系统抽象层，让你可以用统一的接口操作不同的文件系统。

## VFS 虚拟文件系统

### VFS 的核心数据结构

```c
// 超级块：描述整个文件系统
struct super_block {
    struct list_head s_list;          // 超级块链表
    dev_t s_dev;                     // 设备号
    unsigned long s_blocksize;        // 块大小
    unsigned char s_blocksize_bits;   // 块大小位数
    void* s_fs_info;                  // 指向具体文件系统的私有数据
    
    struct file_system_type* s_type; // 文件系统类型
    struct super_operations* s_op;    // 超级块操作
    
    struct dentry* s_root;            // 根目录项
    struct quota_info s_dquot;        // 磁盘配额
    
    // ... 其他字段
};

// inode：描述文件
struct inode {
    umode_t i_mode;                   // 文件类型和权限
    unsigned int i_flags;            // 文件标志
    loff_t i_size;                   // 文件大小
    unsigned long i_blocks;          // 文件占用的块数
    struct timespec i_atime;         // 最后访问时间
    struct timespec i_mtime;         // 最后修改时间
    struct timespec i_ctime;         // 最后状态改变时间
    
    struct inode_operations* i_op;    // inode 操作
    struct file_operations* i_fop;    // 文件操作
    struct super_block* i_sb;         // 所属超级块
    
    void* i_private;                  // 指向具体文件系统的私有数据
    
    // ... 其他字段
};

// 文件：描述进程打开的文件
struct file {
    union {
        struct llist_node fu_llist;
        struct rcu_head fu_rcuhead;
    } f_u;
    struct path f_path;               // 文件路径
    struct inode* f_inode;           // 对应的 inode
    const struct file_operations* f_op; // 文件操作
    
    loff_t f_pos;                    // 文件位置
    unsigned int f_flags;            // 文件标志
    fmode_t f_mode;                  // 文件模式
    
    // ... 其他字段
};
```

### 目录项缓存（dentry cache）

VFS 使用 dentry cache 加速路径解析：

```c
// 目录项：描述文件路径中的每一个组件
struct dentry {
    unsigned int d_flags;            // 目录项标志
    seqcount_t d_seq;                // 序列号
    struct hlist_node d_hash;        // 哈希表链表
    struct dentry* d_parent;         // 父目录
    struct qstr d_name;              // 文件名
    unsigned char d_iname[DNAME_INLINE_LEN]; // 短文件名
    
    struct inode* d_inode;           // 目录项对应的 inode
    struct dentry_operations* d_op; // 目录项操作
    
    struct super_block* d_sb;        // 所属超级块
    unsigned long d_time;             // 时间
    void* d_fsdata;                  // 具体文件系统的私有数据
    
    struct list_head d_child;        // 在父目录的子链表
    struct list_head d_subdirs;      // 子目录链表
};
```

```
路径解析示例: /home/user/file.txt

/home/user/file.txt
 │
 ▼
 dentry: "/"        → inode: root (/)
 │
 ▼
 dentry: "home"     → inode: /home 目录
 │
 ▼
 dentry: "user"     → inode: /home/user 目录
 │
 ▼
 dentry: "file.txt" → inode: file.txt 文件
```

## Ext4 文件系统

### Ext4 的结构

```
┌─────────────────────────────────────────────────────────────────┐
│                        Ext4 文件系统布局                          │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  块组 (Block Group):                                            │
│  ┌──────────┬──────────┬──────────┬──────────┬──────────┐       │
│  │ Super    │  GDT     │  Block   │  Inode   │  Data    │       │
│  │ Block    │  Backup  │  Bitmap  │  Bitmap  │  Blocks  │       │
│  │          │          │          │          │          │       │
│  └──────────┴──────────┴──────────┴──────────┴──────────┘       │
│                                                                 │
│  特点:                                                          │
│  - 每个块组有备份的超级块和 GDT（冗余）                          │
│  - 块组减少碎片，提高性能                                        │
│  - 延迟分配，优化写入顺序                                        │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### Ext4 的 inode

```c
// Ext4 inode 结构
struct ext4_inode {
    __le16  i_mode;                  // 文件类型和权限
    __le16  i_uid;                   // UID
    __le32  i_size_lo;               // 文件大小（低 32 位）
    __le32  i_atime;                // 访问时间
    __le32  i_ctime;                // 创建时间
    __le32  i_mtime;                // 修改时间
    __le32  i_dtime;                // 删除时间
    __le16  i_gid;                   // GID
    __le16  i_links_count;           // 硬链接计数
    __le32  i_blocks_lo;             // 占用的块数
    __le32  i_flags;                // 文件标志
    __le32  i_osd1;                 // OS 相关
    __le32  i_block[15];            // 数据块指针！！！
    __le32  i_generation;            // 文件版本（NFS）
    __le32  i_file_acl_lo;          // ACL
    __le32  i_size_high;             // 文件大小（高 32 位）
    __le32  i_fragment_addr;         // 片段地址
    __le32  i_osd2[3];              // OS 相关
};
```

### Ext4 的 extent 树

Ext4 使用 extent 代替传统的块指针，提高大文件效率：

```c
// Ext4 extent 头
struct ext4_extent_header {
    __le16  eh_magic;               // 魔数 (0xF30A)
    __le16  eh_entries;             // extent 数量
    __le16  eh_max;                 // 最大 extent 数
    __le16  eh_depth;               // extent 树深度
    __le32  eh_generation;          // 生成号
};

// Ext4 extent 项
struct ext4_extent {
    __le32  ee_block;               // 逻辑块号
    __le16  ee_len;                 // 物理块数量
    __le16  ee_start_hi;           // 物理块号（高 16 位）
    __le32  ee_start_lo;           // 物理块号（低 32 位）
};
```

```
extent vs 传统块指针:

传统方式 (i_block[15]):
┌─────────────────────────────────────────────────────────────┐
│  12 个直接块指针                                            │
│  1 个单间接块指针 (1024 个块)                                │
│  1 个双间接块指针                                           │
│  1 个三间接块指针                                           │
└─────────────────────────────────────────────────────────────┘

extent 方式:
┌─────────────────────────────────────────────────────────────┐
│  extent[0]: 逻辑块 0-100 → 物理块 500-600                   │
│  extent[1]: 逻辑块 101-200 → 物理块 1000-1099                │
│  extent[2]: 逻辑块 201-256 → 物理块 2000-2055                │
│                                                              │
│  优点: 存储效率高，大文件 I/O 次数少                         │
└─────────────────────────────────────────────────────────────┘
```

### Ext4 的日志

```c
// Ext4 日志（journal）模式:
// - data=writeback: 只记录元数据
```

## 面试高频问题

**Q1：VFS 是什么？**

VFS（Virtual File System）是 Linux 的虚拟文件系统层，向上提供统一的文件操作接口（open、read、write 等），向下支持多种具体文件系统（ext4、XFS、Btrfs 等）。VFS 让应用程序可以用相同的方式操作不同的文件系统。

**Q2：Ext4 的 extent 有什么优点？**

extent 用一个 (逻辑块, 物理块, 长度) 描述连续的数据块，减少了 inode 中块指针的数量。对于连续的大文件，只需要一个 extent，避免了多级间接块查找，提高了 I/O 效率。

**Q3：dentry cache 的作用是什么？**

目录项缓存加速路径解析。当访问 /home/user/file.txt 时，内核会创建相应的 dentry 节点并缓存。再次访问时，可以直接从缓存获取，避免重复解析。

**Q4：Ext4 和 XFS 的区别？**

Ext4 适合中小文件，兼容性更好，支持配额和 ACL。XFS 适合大文件和大容量存储，元数据操作更高效，支持更大的文件和文件系统。

**Q5：什么是日志文件系统？**

日志文件系统在写入正式数据前，先把操作记录到日志区。如果写入过程崩溃，恢复时只需检查并重放日志，不用全盘扫描。Ext4、XFS、NTFS 都是日志文件系统。

理解 Linux 文件系统，是理解 Linux 存储子系统的关键。
