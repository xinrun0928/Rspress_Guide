# RBAC 权限模型

你的团队有 10 个人，每个人都负责不同的模块。如果直接给每个人分配权限，当有人离职或调岗时，你需要逐个修改权限。

**RBAC（Role-Based Access Control）** 通过引入「角色」这个中间层，简化了权限管理。

## RBAC 核心概念

```
用户 (User) -> 分配 -> 角色 (Role) -> 包含 -> 权限 (Permission)

用户：张三
角色：管理员
权限：用户管理、数据删除、系统配置
```

### 基本元素

| 元素 | 说明 |
|---|---|
| 用户（User） | 系统的使用者 |
| 角色（Role） | 一组权限的集合，代表一个职责 |
| 权限（Permission） | 对资源的操作许可 |
| 资源（Resource） | 被保护的对象（文件、API、数据） |
| 操作（Operation） | 对资源的动作（读、写、删） |

## RBAC 模型

### RBAC0：基础模型

最简单的情况，用户、角色、权限之间是多对多关系：

```python
# RBAC0 数据模型
class RBAC:
    def __init__(self):
        # 用户表
        self.users = {}  # user_id -> {username, ...}

        # 角色表
        self.roles = {}  # role_id -> {role_name, ...}

        # 权限表
        self.permissions = {}  # perm_id -> {resource, operation}

        # 用户-角色关联表
        self.user_roles = {}  # user_id -> [role_id, ...]

        # 角色-权限关联表
        self.role_permissions = {}  # role_id -> [perm_id, ...]

    def assign_role(self, user_id, role_id):
        """分配角色给用户"""
        if user_id not in self.user_roles:
            self.user_roles[user_id] = []
        self.user_roles[user_id].append(role_id)

    def grant_permission(self, role_id, permission_id):
        """给角色授权"""
        if role_id not in self.role_permissions:
            self.role_permissions[role_id] = []
        self.role_permissions[role_id].append(permission_id)

    def check_permission(self, user_id, resource, operation):
        """检查用户是否有权限"""
        # 获取用户的所有角色
        roles = self.user_roles.get(user_id, [])

        # 检查每个角色是否有对应权限
        for role_id in roles:
            perms = self.role_permissions.get(role_id, [])
            for perm_id in perms:
                perm = self.permissions.get(perm_id)
                if perm and perm['resource'] == resource and perm['operation'] == operation:
                    return True

        return False
```

### RBAC1：角色分层

角色之间有层级关系，高级角色继承低级角色的权限：

```python
# RBAC1：角色分层
class RBAC1(RBAC):
    def __init__(self):
        super().__init__()
        # 角色层级关系
        self.role_hierarchy = {}  # role_id -> parent_role_id

    def assign_role(self, user_id, role_id):
        """分配角色，自动继承父角色"""
        super().assign_role(user_id, role_id)

        # 自动继承父角色的权限
        current_role = role_id
        while current_role in self.role_hierarchy:
            parent_role = self.role_hierarchy[current_role]
            super().grant_permission(role_id, self.role_permissions.get(parent_role, []))
            current_role = parent_role

    def check_permission(self, user_id, resource, operation):
        """检查权限，包括继承的权限"""
        roles = self.user_roles.get(user_id, [])

        for role_id in roles:
            if self._check_role_permission(role_id, resource, operation):
                return True

        return False

    def _check_role_permission(self, role_id, resource, operation):
        """递归检查角色及其父角色的权限"""
        perms = self.role_permissions.get(role_id, [])
        for perm_id in perms:
            perm = self.permissions.get(perm_id)
            if perm and perm['resource'] == resource and perm['operation'] == operation:
                return True

        # 检查父角色
        if role_id in self.role_hierarchy:
            return self._check_role_permission(self.role_hierarchy[role_id], resource, operation)

        return False
```

```
角色层级示例：

        CEO
       /   \
    CTO    CFO
    /  \     \
  PM   DEV   FIN
   |
  QA

QA -> PM -> CTO -> CEO
QA 继承 PM 的权限，PM 继承 CTO 的权限
```

### RBAC2：角色约束

添加约束条件，防止权限冲突：

| 约束类型 | 说明 | 例子 |
|---|---|---|
| 互斥角色 | 用户不能同时拥有两个互斥角色 | 会计 vs 出纳 |
| 基数约束 | 限制角色分配的数量 | 一个角色最多 10 个用户 |
| 先决条件 | 必须拥有某个角色才能获得另一个 | 先是 Developer 才能是 Tech Lead |

```python
# RBAC2：添加约束
class RBAC2(RBAC):
    def __init__(self):
        super().__init__()
        # 互斥角色组
        self.mutually_exclusive_roles = {}  # role_id -> [互斥角色列表]

        # 角色基数
        self.role_max_users = {}  # role_id -> 最大用户数

        # 先决角色
        self.prerequisite_roles = {}  # role_id -> [必须先拥有的角色]

    def assign_role(self, user_id, role_id):
        """分配角色（带约束检查）"""
        # 1. 检查互斥角色
        if role_id in self.mutually_exclusive_roles:
            user_roles = set(self.user_roles.get(user_id, []))
            exclusive = set(self.mutually_exclusive_roles[role_id])
            if user_roles & exclusive:  # 有交集，说明违反了互斥
                raise PermissionError("互斥角色约束")

        # 2. 检查基数约束
        if role_id in self.role_max_users:
            current_users = sum(1 for users in self.user_roles.values() if role_id in users)
            if current_users >= self.role_max_users[role_id]:
                raise PermissionError("角色用户数已达上限")

        # 3. 检查先决条件
        if role_id in self.prerequisite_roles:
            user_roles = set(self.user_roles.get(user_id, []))
            required = set(self.prerequisite_roles[role_id])
            if not required.issubset(user_roles):
                raise PermissionError("缺少先决角色")

        super().assign_role(user_id, role_id)
```

### RBAC3：完整模型

RBAC3 = RBAC0 + RBAC1 + RBAC2，包含所有特性。

## 数据库设计

```sql
-- 用户表
CREATE TABLE users (
    id INT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 角色表
CREATE TABLE roles (
    id INT PRIMARY KEY,
    name VARCHAR(50) UNIQUE NOT NULL,
    description VARCHAR(255),
    parent_id INT REFERENCES roles(id)  -- 角色层级
);

-- 权限表
CREATE TABLE permissions (
    id INT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    resource VARCHAR(100) NOT NULL,   -- 资源，如 'user', 'order', '/api/admin/*'
    operation VARCHAR(50) NOT NULL,    -- 操作，如 'read', 'write', 'delete'
    description VARCHAR(255)
);

-- 用户-角色关联表
CREATE TABLE user_roles (
    user_id INT REFERENCES users(id),
    role_id INT REFERENCES roles(id),
    PRIMARY KEY (user_id, role_id)
);

-- 角色-权限关联表
CREATE TABLE role_permissions (
    role_id INT REFERENCES roles(id),
    permission_id INT REFERENCES permissions(id),
    PRIMARY KEY (role_id, permission_id)
);

-- 互斥角色约束表
CREATE TABLE role_mutex (
    role_id1 INT REFERENCES roles(id),
    role_id2 INT REFERENCES roles(id),
    PRIMARY KEY (role_id1, role_id2)
);
```

## 实际应用

### Python 实现

```python
from sqlalchemy import Column, Integer, String, ForeignKey, Table
from sqlalchemy.orm import relationship, declarative_base

Base = declarative_base()

# 关联表
user_roles = Table('user_roles', Base.metadata,
    Column('user_id', Integer, ForeignKey('users.id'), primary_key=True),
    Column('role_id', Integer, ForeignKey('roles.id'), primary_key=True)
)

role_permissions = Table('role_permissions', Base.metadata,
    Column('role_id', Integer, ForeignKey('roles.id'), primary_key=True),
    Column('permission_id', Integer, ForeignKey('permissions.id'), primary_key=True)
)

class User(Base):
    __tablename__ = 'users'
    id = Column(Integer, primary_key=True)
    username = Column(String(50), unique=True)

    roles = relationship('Role', secondary=user_roles, back_populates='users')

class Role(Base):
    __tablename__ = 'roles'
    id = Column(Integer, primary_key=True)
    name = Column(String(50), unique=True)
    parent_id = Column(Integer, ForeignKey('roles.id'))

    users = relationship('User', secondary=user_roles, back_populates='roles')
    permissions = relationship('Permission', secondary=role_permissions)

class Permission(Base):
    __tablename__ = 'permissions'
    id = Column(Integer, primary_key=True)
    name = Column(String(50))
    resource = Column(String(100))
    operation = Column(String(50))

# 权限检查装饰器
from functools import wraps
from flask import abort

def require_permission(resource, operation):
    def decorator(f):
        @wraps(f)
        def decorated_function(*args, **kwargs):
            if not current_user.has_permission(resource, operation):
                abort(403)  # Forbidden
            return f(*args, **kwargs)
        return decorated_function
    return decorator

# 使用
@app.route('/admin/users')
@login_required
@require_permission('user', 'read')
def list_users():
    return render_template('users.html')
```

### 微服务权限控制

```python
# 微服务中的权限控制
class PermissionService:
    def __init__(self, redis_client):
        self.redis = redis_client
        self.cache_ttl = 300  # 缓存 5 分钟

    def check_permission(self, user_id, resource, operation):
        """检查用户权限"""
        cache_key = f"perm:{user_id}"

        # 先查缓存
        cached = self.redis.get(cache_key)
        if cached:
            permissions = json.loads(cached)
        else:
            # 查数据库
            permissions = self._load_permissions(user_id)
            self.redis.setex(cache_key, self.cache_ttl, json.dumps(permissions))

        return (resource, operation) in permissions

    def clear_permission_cache(self, user_id):
        """用户权限变更时清除缓存"""
        self.redis.delete(f"perm:{user_id}")

# API 网关中的权限验证
class APIGateway:
    def __init__(self, permission_service):
        self.permission_service = permission_service

    async def handle_request(self, request):
        user = request.user
        path = request.path
        method = request.method

        # 从路径提取资源
        resource = self._extract_resource(path)

        if not self.permission_service.check_permission(user.id, resource, method):
            return Response(status=403, body="Forbidden")

        return await self.forward_request(request)
```

## 与其他模型对比

| 模型 | 说明 | 适用场景 |
|---|---|---|
| ACL | 直接给用户分配权限 | 小型系统 |
| RBAC | 通过角色管理权限 | 中型企业 |
| ABAC | 基于属性的权限控制 | 复杂业务规则 |
| PBAC | 基于策略的访问控制 | 灵活策略管理 |

### ABAC 示例

```python
# ABAC：基于属性的访问控制
class ABAC:
    def can_access(self, subject, resource, action, environment):
        """
        subject: 用户的属性
        resource: 资源的属性
        action: 操作
        environment: 环境属性（时间、位置等）
        """
        # 策略：只有文件所有者或同部门的人才能读写
        rules = [
            {
                "effect": "permit",
                "condition": lambda s, r, a, e: (
                    s.id == r.owner_id or
                    s.department == r.department
                )
            },
            {
                "effect": "deny",
                "condition": lambda s, r, a, e: (
                    e.time.hour < 9 or e.time.hour > 18  # 非工作时间
                )
            }
        ]

        for rule in rules:
            if rule['condition'](subject, resource, action, environment):
                return rule['effect'] == 'permit'

        return False
```
