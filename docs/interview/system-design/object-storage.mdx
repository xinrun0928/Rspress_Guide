# 对象存储系统设计

你有没有想过：

- 阿里云 OSS、AWS S3 是怎么存储几十亿张图片的？
- 为什么上传一张图片后，任何地方都能访问？
- 为什么上传失败后，重试可以继续？

这些问题的背后，都是**对象存储系统**在默默工作。

今天，我们来深入探讨对象存储系统的设计与实现。

```java title="src/main/java/DataSharding.java"
/**
 * 数据分片策略
 */
public class DataSharding {
    
    /**
     * 分片存储架构
     */
    public class DataLayout {
        // 单个数据块大小：64MB
        public static final int CHUNK_SIZE = 64 * 1024 * 1024;
        
        // 单节点存储上限：100TB
        public static final long NODE_CAPACITY = 100L * 1024 * 1024 * 1024 * 1024;
        
        // 分片数：N 个副本
        public static final int REPLICA_COUNT = 3;
    }
    
    /**
     * 选择存储节点
     */
    public List<DataNode> selectNodes(String bucket, String key) {
        // 1. 计算 hash
        int hash = Math.abs(key.hashCode());
        
        // 2. 一致性哈希选择主节点
        DataNode primary = consistentHash.getNode(hash);
        
        // 3. 选择从节点（同机房优先）
        List<DataNode> replicas = selectReplicas(primary, REPLICA_COUNT - 1);
        
        // 4. 返回所有节点
        List<DataNode> result = new ArrayList<>();
        result.add(primary);
        result.addAll(replicas);
        return result;
    }
    
    /**
     * 一致性哈希选择节点
     */
    public class ConsistentHashSelector {
        private TreeMap<Long, DataNode> ring;
        
        public DataNode getNode(int hash) {
            // 找到第一个大于等于 hash 的节点
            Map.Entry<Long, DataNode> entry = ring.ceilingEntry((long) hash);
            
            if (entry == null) {
                entry = ring.firstEntry();
            }
            
            return entry.getValue();
        }
    }
}
```

```java title="src/main/java/MultipartUpload.java"
/**
 * 分片上传流程
 */
public class MultipartUpload {
    
    /**
     * 上传流程
     */
    public void upload(File file) {
        // 1. 创建上传任务
        String uploadId = createUploadTask();
        
        // 2. 计算分片
        int partSize = 5 * 1024 * 1024; // 5MB
        List<Part> parts = splitFile(file, partSize);
        
        // 3. 上传分片（并行）
        ExecutorService executor = Executors.newFixedThreadPool(10);
        List<Future<PartETag>> futures = new ArrayList<>();
        
        for (Part part : parts) {
            Future<PartETag> future = executor.submit(() -> {
                return uploadPart(uploadId, part);
            });
            futures.add(future);
        }
        
        // 4. 收集结果
        List<PartETag> partETags = new ArrayList<>();
        for (Future<PartETag> future : futures) {
            partETags.add(future.get());
        }
        
        // 5. 完成上传
        completeUpload(uploadId, partETags);
    }
    
    /**
     * 断点续传
     */
    public void resumeUpload(String uploadId) {
        // 1. 查询已上传的分片
        List<PartETag> uploadedParts = listParts(uploadId);
        
        // 2. 获取未完成的部分
        List<Part> remainingParts = getRemainingParts(uploadId, uploadedParts);
        
        // 3. 继续上传
        for (Part part : remainingParts) {
            uploadPart(uploadId, part);
        }
        
        // 4. 完成上传
        completeUpload(uploadId, uploadedParts);
    }
}
```

```java title="src/main/java/ReplicaStrategy.java"
/**
 * 多副本策略
 */
public class ReplicaStrategy {
    
    /**
     * 副本写入流程
     */
    public void writeWithReplicas(Data data, List<DataNode> nodes) {
        // 1. 写入所有副本
        List<CompletableFuture<Boolean>> futures = new ArrayList<>();
        
        for (DataNode node : nodes) {
            CompletableFuture<Boolean> future = CompletableFuture.supplyAsync(() -> {
                return writeToNode(node, data);
            });
            futures.add(future);
        }
        
        // 2. 等待写入完成（多数派确认即可）
        int successCount = 0;
        for (CompletableFuture<Boolean> future : futures) {
            if (future.join()) {
                successCount++;
            }
        }
        
        // 3. 判断是否成功
        if (successCount >= nodes.size() / 2 + 1) {
            return; // 成功
        } else {
            throw new WriteFailedException();
        }
    }
    
    /**
     * 副本修复
     */
    public void repairReplica(DataChunk chunk) {
        // 1. 检查所有副本
        List<ReplicaStatus> statuses = checkAllReplicas(chunk);
        
        // 2. 找到完整副本
        DataNode sourceNode = statuses.stream()
            .filter(ReplicaStatus::isHealthy)
            .findFirst()
            .orElseThrow(() -> new NoHealthyReplicaException());
        
        // 3. 修复损坏副本
        for (ReplicaStatus status : statuses) {
            if (!status.isHealthy()) {
                copyFromNode(sourceNode, status.getNode(), chunk);
            }
        }
    }
}
```

```java title="src/main/java/DataRepair.java"
/**
 * 数据修复
 */
public class DataRepair {
    
    /**
     * 完整性检查
     */
    public void checkIntegrity() {
        // 1. 扫描所有数据块
        for (DataChunk chunk : chunkScanner.scanAll()) {
            // 2. 验证 checksum
            if (!verifyChecksum(chunk)) {
                // 3. 触发修复
                repairChunk(chunk);
            }
        }
    }
    
    /**
     * CRC 校验
     */
    public boolean verifyChecksum(DataChunk chunk) {
        CRC32 crc32 = new CRC32();
        crc32.update(chunk.getData());
        long expected = crc32.getValue();
        return expected == chunk.getChecksum();
    }
    
    /**
     * 副本一致性修复
     */
    public void repairInconsistency(String objectKey) {
        // 1. 获取对象元数据
        ObjectMetadata meta = metadataService.getObjectMetadata(objectKey);
        
        // 2. 检查所有副本
        List<ReplicaData> replicas = replicaManager.getReplicas(objectKey);
        
        // 3. 找到最新版本
        ReplicaData latest = replicas.stream()
            .max(Comparator.comparing(ReplicaData::getVersion))
            .orElseThrow(() -> new NoReplicaException());
        
        // 4. 同步到其他副本
        for (ReplicaData replica : replicas) {
            if (!replica.getVersion().equals(latest.getVersion())) {
                syncToReplica(replica, latest);
            }
        }
    }
}
```

```java title="src/main/java/SignatureAuth.java"
/**
 * 签名认证
 */
public class SignatureAuth {
    
    /**
     * 生成签名
     */
    public String generateSignature(String secretKey, String method, 
                                   String path, long timestamp) {
        // 1. 构建签名字符串
        String signString = method + "\n" 
                          + path + "\n" 
                          + timestamp;
        
        // 2. HMAC-SHA1 签名
        Mac mac = Mac.getInstance("HmacSHA1");
        SecretKeySpec keySpec = new SecretKeySpec(
            secretKey.getBytes(), "HmacSHA1");
        mac.init(keySpec);
        byte[] hash = mac.doFinal(signString.getBytes());
        
        // 3. Base64 编码
        return Base64.encode(hash);
    }
    
    /**
     * 验证签名
     */
    public boolean verifySignature(Request request) {
        String accessKey = request.getHeader("X-Access-Key");
        String signature = request.getHeader("X-Signature");
        String timestamp = request.getHeader("X-Timestamp");
        
        // 1. 检查时间戳（5 分钟内有效）
        long requestTime = Long.parseLong(timestamp);
        if (Math.abs(System.currentTimeMillis() - requestTime) > 5 * 60 * 1000) {
            return false;
        }
        
        // 2. 获取 SecretKey
        String secretKey = getSecretKey(accessKey);
        
        // 3. 重新计算签名并比对
        String expected = generateSignature(secretKey, 
            request.getMethod(), request.getPath(), requestTime);
        
        return expected.equals(signature);
    }
}
```

```java title="src/main/java/LifecycleManagement.java"
    /**
     * 配置生命周期规则
     */
    public void configureLifecycle(String bucket) {
        LifecycleRule rule = new LifecycleRule();
        
        // 1. 30 天后转标准存储
        rule.addTransition(new Transition()
            .withDays(30)
            .withStorageClass(StorageClass.StandardIA));
        
        // 2. 90 天后转归档存储
        rule.addTransition(new Transition()
            .withDays(90)
            .withStorageClass(StorageClass.Archive));
        
        // 3. 365 天后删除
        rule.addExpiration(new Expiration()
            .withDays(365));
        
        ossClient.setBucketLifecycle(bucket, Collections.singletonList(rule));
    }
}
```

### 问题一：「如何保证数据不丢失？」

**回答思路**：

```
方案：
1. 多副本存储：至少 3 副本
2. 纠删码：类似 RAID，容错率更高
3. 跨机房复制：防止机房级故障
4. 定期校验：检查数据完整性
```

**回答思路**：

```
方案：
1. CDN 加速：边缘节点缓存
2. 热点缓存：热点数据预加载到内存
3. 负载均衡：多节点并行读取
4. 并行下载：大文件分片并行
```

### 问题三：「如何设计一个 S3？」

**回答思路**：

```
核心组件：
1. API 网关：签名、路由、限流
2. 元数据存储：对象信息、Bucket 信息
3. 数据存储节点：数据分片、多副本
4. 一致性哈希：数据定位

关键设计：
1. 扁平的命名空间
2. HTTP API
3. 多副本强一致
4. 签名认证
```
