# 消息队列与粉丝关系

Redis 的 List 可以实现简单的消息队列，Set 可以实现粉丝关系等功能。这些是 Redis 的经典应用场景。

## List 实现消息队列

### 基础操作

```bash
# 生产者：向队列添加消息
LPUSH queue:orders "order:1001"
LPUSH queue:orders "order:1002"
LPUSH queue:orders "order:1003"

# 消费者：阻塞式消费
BRPOP queue:orders 0
# 返回: ["queue:orders", "order:1001"]
```

### 生产者-消费者模式

```java
@Service
public class OrderMessageQueue {
    
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    
    private static final String QUEUE_KEY = "queue:orders";
    
    /**
     * 生产消息
     */
    public void sendOrder(Long orderId) {
        String message = JSON.toJSONString(new OrderMessage(orderId, System.currentTimeMillis()));
        redisTemplate.opsForList().leftPush(QUEUE_KEY, message);
    }
    
    /**
     * 消费消息
     */
    public OrderMessage receiveOrder() {
        String message = redisTemplate.opsForList().rightPop(QUEUE_KEY);
        if (message != null) {
            return JSON.parseObject(message, OrderMessage.class);
        }
        return null;
    }
    
    /**
     * 阻塞消费
     */
    public OrderMessage blockingReceiveOrder() {
        // BRPOP：阻塞直到有消息
        List<String> result = redisTemplate.opsForList().rightPop(
            QUEUE_KEY, 0, TimeUnit.SECONDS);
        
        if (result != null && !result.isEmpty()) {
            return JSON.parseObject(result.get(1), OrderMessage.class);
        }
        return null;
    }
}
```

### 消费者处理

```java
@Service
public class OrderConsumer {
    
    @Autowired
    private OrderMessageQueue messageQueue;
    
    @Autowired
    private OrderService orderService;
    
    /**
     * 启动消费者
     */
    @PostConstruct
    public void startConsumer() {
        Thread consumerThread = new Thread(() -> {
            while (true) {
                try {
                    OrderMessage message = messageQueue.blockingReceiveOrder();
                    if (message != null) {
                        processOrder(message);
                    }
                } catch (Exception e) {
                    log.error("处理订单失败", e);
                }
            }
        });
        consumerThread.setName("order-consumer");
        consumerThread.start();
    }
    
    private void processOrder(OrderMessage message) {
        log.info("处理订单: {}", message.getOrderId());
        orderService.processOrder(message.getOrderId());
    }
}
```

### 消息确认机制

```java
@Service
public class ReliableMessageQueue {
    
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    
    private static final String PROCESSING_QUEUE = "queue:orders:processing";
    private static final String READY_QUEUE = "queue:orders:ready";
    
    /**
     * 发送消息
     */
    public void sendMessage(String message) {
        redisTemplate.opsForList().leftPush(READY_QUEUE, message);
    }
    
    /**
     * 消费消息（带确认）
     */
    public String receiveMessage() {
        // 从就绪队列取出
        String message = redisTemplate.opsForList().rightPop(READY_QUEUE);
        
        if (message != null) {
            // 放入处理中队列
            redisTemplate.opsForList().leftPush(PROCESSING_QUEUE, message);
        }
        
        return message;
    }
    
    /**
     * 确认消息处理成功
     */
    public void ackMessage(String message) {
        // 从处理中队列移除
        redisTemplate.opsForList().remove(PROCESSING_QUEUE, 1, message);
    }
    
    /**
     * 重试失败消息
     */
    public void retryMessage() {
        // 将处理中的消息放回就绪队列
        String message = redisTemplate.opsForList().rightPop(PROCESSING_QUEUE);
        if (message != null) {
            redisTemplate.opsForList().leftPush(READY_QUEUE, message);
        }
    }
}
```

## Set 实现粉丝关系

### 基本操作

```bash
# 用户 123 的粉丝
SADD user:123:followers 456 789 101

# 用户 123 关注的用户
SADD user:123:following 100 200 300

# 查询用户 123 是否关注了用户 456
SISMEMBER user:123:following 456
# (integer) 0  # 没关注

# 获取用户 123 的关注列表
SMEMBERS user:123:following
# 1) "100"
# 2) "200"
# 3) "300"

# 获取共同关注
SINTER user:123:following user:456:following
```

### 粉丝关系服务

```java
@Service
public class FollowService {
    
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    
    /**
     * 关注用户
     */
    public void follow(Long userId, Long targetUserId) {
        String followingKey = "user:" + userId + ":following";
        String followerKey = "user:" + targetUserId + ":followers";
        
        // 我关注的人 +1
        redisTemplate.opsForSet().add(followingKey, targetUserId.toString());
        // 对方的粉丝 +1
        redisTemplate.opsForSet().add(followerKey, userId.toString());
    }
    
    /**
     * 取消关注
     */
    public void unfollow(Long userId, Long targetUserId) {
        String followingKey = "user:" + userId + ":following";
        String followerKey = "user:" + targetUserId + ":followers";
        
        redisTemplate.opsForSet().remove(followingKey, targetUserId.toString());
        redisTemplate.opsForSet().remove(followerKey, userId.toString());
    }
    
    /**
     * 检查是否关注
     */
    public boolean isFollowing(Long userId, Long targetUserId) {
        String followingKey = "user:" + userId + ":following";
        return redisTemplate.opsForSet().isMember(followingKey, targetUserId.toString());
    }
    
    /**
     * 获取关注列表
     */
    public Set<Long> getFollowing(Long userId) {
        String followingKey = "user:" + userId + ":following";
        Set<String> members = redisTemplate.opsForSet().members(followingKey);
        return members.stream().map(Long::parseLong).collect(Collectors.toSet());
    }
    
    /**
     * 获取粉丝列表
     */
    public Set<Long> getFollowers(Long userId) {
        String followerKey = "user:" + userId + ":followers";
        Set<String> members = redisTemplate.opsForSet().members(followerKey);
        return members.stream().map(Long::parseLong).collect(Collectors.toSet());
    }
    
    /**
     * 获取粉丝数量
     */
    public Long getFollowerCount(Long userId) {
        String followerKey = "user:" + userId + ":followers";
        return redisTemplate.opsForSet().size(followerKey);
    }
    
    /**
     * 获取关注数量
     */
    public Long getFollowingCount(Long userId) {
        String followingKey = "user:" + userId + ":following";
        return redisTemplate.opsForSet().size(followingKey);
    }
    
    /**
     * 获取共同关注
     */
    public Set<Long> getCommonFollowing(Long userId1, Long userId2) {
        String followingKey1 = "user:" + userId1 + ":following";
        String followingKey2 = "user:" + userId2 + ":following";
        
        Set<String> common = redisTemplate.opsForSet().intersect(followingKey1, followingKey2);
        return common.stream().map(Long::parseLong).collect(Collectors.toSet());
    }
}
```

## 粉丝推送（Feed）

当用户发布内容时，推送给所有粉丝：

```java
@Service
public class FeedService {
    
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    
    @Autowired
    private FollowService followService;
    
    /**
     * 发布动态
     */
    public void publish(Long userId, String content) {
        String postId = UUID.randomUUID().toString();
        
        // 保存帖子
        String postKey = "post:" + postId;
        Map<String, String> post = new HashMap<>();
        post.put("id", postId);
        post.put("userId", userId.toString());
        post.put("content", content);
        post.put("timestamp", String.valueOf(System.currentTimeMillis()));
        redisTemplate.opsForHash().putAll(postKey, post);
        
        // 获取粉丝列表
        Set<Long> followers = followService.getFollowers(userId);
        
        // 推送给所有粉丝（写入他们的 timeline）
        for (Long followerId : followers) {
            String timelineKey = "timeline:" + followerId;
            // ZSet：score 为时间戳，member 为帖子 ID
            redisTemplate.opsForZSet().add(timelineKey, postId, System.currentTimeMillis());
            
            // 限制 timeline 大小
            redisTemplate.opsForZSet().removeRange(timelineKey, 0, -1001);
        }
    }
    
    /**
     * 获取用户的 timeline
     */
    public List<Map<String, Object>> getTimeline(Long userId, int page, int size) {
        String timelineKey = "timeline:" + userId;
        
        // 按时间倒序获取
        Set<String> postIds = redisTemplate.opsForZSet()
            .reverseRange(timelineKey, page * sizeL, (page + 1) * sizeL - 1);
        
        List<Map<String, Object>> posts = new ArrayList<>();
        for (String postId : postIds) {
            Map<Object, Object> post = redisTemplate.opsForHash().entries("post:" + postId);
            if (!post.isEmpty()) {
                posts.add(post);
            }
        }
        
        return posts;
    }
}
```

## Set 的其他应用

### 标签系统

```java
@Service
public class TagService {
    
    /**
     * 给文章添加标签
     */
    public void addTags(Long articleId, String... tags) {
        String tagKey = "article:" + articleId + ":tags";
        redisTemplate.opsForSet().add(tagKey, tags);
    }
    
    /**
     * 获取文章的所有标签
     */
    public Set<String> getArticleTags(Long articleId) {
        String tagKey = "article:" + articleId + ":tags";
        return redisTemplate.opsForSet().members(tagKey);
    }
    
    /**
     * 获取有共同标签的文章
     */
    public Set<Long> getArticlesByTags(String... tags) {
        String tagKey1 = "tag:" + tags[0] + ":articles";
        
        if (tags.length == 1) {
            return redisTemplate.opsForSet().members(tagKey1)
                .stream().map(Long::parseLong).collect(Collectors.toSet());
        }
        
        // 多个标签：求交集
        String tagKey2 = "tag:" + tags[1] + ":articles";
        Set<String> result = redisTemplate.opsForSet().intersect(tagKey1, tagKey2);
        
        for (int i = 2; i < tags.length; i++) {
            String tagKey = "tag:" + tags[i] + ":articles";
            result = redisTemplate.opsForSet().intersect(result, tagKey);
        }
        
        return result.stream().map(Long::parseLong).collect(Collectors.toSet());
    }
}
```

### 抽奖系统

```java
@Service
public class LotteryService {
    
    /**
     * 参与抽奖
     */
    public void participate(Long activityId, Long userId) {
        String key = "lottery:" + activityId + ":participants";
        redisTemplate.opsForSet().add(key, userId.toString());
    }
    
    /**
     * 抽取中奖者
     */
    public List<Long> drawWinners(Long activityId, int count) {
        String key = "lottery:" + activityId + ":participants";
        
        // 随机抽取指定数量的参与者
        Set<String> winners = redisTemplate.opsForSet().distinctRandomMembers(key, count);
        
        // 删除已中奖的参与者（防止重复中奖）
        if (winners != null) {
            redisTemplate.opsForSet().remove(key, winners.toArray());
        }
        
        return winners.stream().map(Long::parseLong).collect(Collectors.toList());
    }
}
```

## List vs Stream

Redis 5.0 引入了 Stream，是一种更完善的消息队列实现：

```bash
# 创建 Stream
XADD mystream * field1 value1 field2 value2

# 读取新消息
XREAD STREAMS mystream $

# 消费者组
XGROUP CREATE mystream mygroup $

# 从消费者组读取
XREADGROUP GROUP mygroup consumer1 STREAMS mystream ">"
```

```java
@Service
public class StreamQueueService {
    
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    
    /**
     * 发送消息到 Stream
     */
    public void sendMessage(String stream, Map<String, String> data) {
        redisTemplate.opsForStream().add(
            new Record.of(stringHashMap(data))
        );
    }
    
    /**
     * 创建消费者组
     */
    public void createGroup(String stream, String group) {
        redisTemplate.opsForStream().createGroup(stream, group);
    }
    
    /**
     * 读取消息
     */
    public List<MapRecord<String, String, String>> readMessage(
            String stream, String group, String consumer) {
        
        List<MapRecord<String, String, String>> records = 
            redisTemplate.opsForStream().read(
                Consumer.from(group, consumer),
                StreamReadOptions.empty().count(1),
                StreamOffset.create(stream, ReadOffset.lastConsumed())
            );
        
        return records;
    }
    
    /**
     * 确认消息
     */
    public void ack(String stream, String group, String recordId) {
        redisTemplate.opsForStream().acknowledge(stream, group, recordId);
    }
}
```

## 总结

List 和 Set 是 Redis 的经典数据结构：

**List 消息队列**：
- LPUSH + BRPOP 实现阻塞队列
- 支持消息确认和重试
- Redis Stream 是更完善的消息队列

**Set 粉丝关系**：
- SADD 添加粉丝/关注
- SISMEMBER 检查是否关注
- SINTER 求共同关注

根据业务需求选择合适的数据结构，才能发挥 Redis 的最大价值。
