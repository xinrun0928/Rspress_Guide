# MongoDB 性能监控

MongoDB 提供了多种工具来监控数据库性能，包括内置命令、工具和 Profiler。

## 监控工具概览

| 工具 | 说明 | 使用场景 |
|---|---|---|
| `db.serverStatus()` | 服务器状态 | 综合监控 |
| `db.collection.stats()` | 集合统计 | 单集合分析 |
| `mongostat` | 实时统计 | 运行时监控 |
| `mongotop` | 操作耗时 | IO 分析 |
| Profiler | 查询分析 | 慢查询追踪 |

## serverStatus

### 基本使用

```javascript
// 查看服务器状态
db.serverStatus()

// 输出包含：
// - connections: 连接数
// - opcounters: 操作计数
// - wiredTiger: 存储引擎统计
// - metrics: 详细指标
// - network: 网络统计
```

### 关键指标

```javascript
// 1. 连接数
db.serverStatus().connections
// { "current": 100, "available": 65436 }

// 2. 操作计数
db.serverStatus().opcounters
// { "insert": 1000, "query": 5000, "update": 2000, "delete": 100 }

// 3. 内存使用
db.serverStatus().wiredTiger.cache
// { "bytes in cache": 6442450944, "percentage used": 75 }

// 4. 复制延迟
db.serverStatus().repl
// { "secondaryDelaySecs": 0.5 }
```

### 定期监控脚本

```javascript
// 监控脚本示例
function monitorStatus() {
  const status = db.serverStatus();

  print("=== MongoDB Status ===");
  print(`Uptime: ${status.uptime} seconds`);
  print(`Connections: ${status.connections.current}/${status.connections.available}`);
  print(`Memory: ${(status.mem.resident / 1024).toFixed(2)} GB`);
  print(`Cache usage: ${status.wiredTiger.cache['percentage of cache in use']}%`);

  print("\n=== Operations (1min) ===");
  print(`Insert: ${status.opcounters.insert}`);
  print(`Query: ${status.opcounters.query}`);
  print(`Update: ${status.opcounters.update}`);
  print(`Delete: ${status.opcounters.delete}`);
}

monitorStatus();
```

## mongostat

### 基本使用

```bash
# 安装 mongostat（随 MongoDB Tools 安装）
# 基本使用
mongostat --host localhost:27017

# 每秒采样
mongostat --host localhost:27017 1

# 指定采样次数
mongostat --host localhost:27017 -n 10 1

# 显示额外指标
mongostat --host localhost:27017 --discover -n 10
```

### 输出说明

```bash
# mongostat 输出
# insert  query update delete getmore command dirty  used flushes  vsize   res qrw  arw  net_in net_out conn   time
#     0     10     0     0       0     1    0.0%  50.0%       0  1.0GB 500MB qrw  arw    500B    1KB   10 12:00:00
#     0      5     2     0       0     2    0.0%  51.0%       0  1.0GB 500MB qrw  arw    400B    1KB   10 12:00:01

# 字段说明：
# - dirty: Cache 中脏页比例
# - used: Cache 使用比例
# - flushes: Checkpoint 次数
# - qrw: 读写队列
# - arw: 活跃读写线程
# - net_in/out: 网络流量
```

## mongotop

### 基本使用

```bash
# 查看各集合的读写时间
mongotop --host localhost:27017

# 每秒采样
mongotop --host localhost:27017 1

# 只显示命名空间
mongotop --host localhost:27017 --namespace shop

# 跳过空闲的命名空间
mongotop --host localhost:27017 --rowcount 10
```

### 输出说明

```bash
# mongotop 输出
#                           ns    total    read    write  2024-01-01T12:00:00
#                    admin.$cmd    0ms     0ms     0ms
#                     shop.orders    15ms    5ms    10ms
#                      shop.users     8ms     8ms     0ms
#                     shop.products    3ms     1ms     2ms

# 字段说明：
# - ns: 命名空间（数据库.集合）
# - total: 总时间
# - read: 读操作时间
# - write: 写操作时间
```

## 集合统计

### stats 命令

```javascript
// 查看集合统计
db.orders.stats()

// 查看集合大小
db.orders.stats().size              // 文档总大小
db.orders.stats().storageSize       // 存储大小（压缩后）
db.orders.stats().totalSize        // 总大小（索引+数据）
db.orders.stats().count            // 文档数

// 查看索引信息
db.orders.stats().indexSizes
// { "_id_": 2097152, "customer_id_1": 3145728 }
```

### 索引使用分析

```javascript
// 查看索引统计
db.orders.aggregate([{ $indexStats: {} }])

// 输出示例
[
  {
    key: { customer_id: 1 },
    name: "customer_id_1",
    accesses: { ops: 1000, since: ISODate("2024-01-01") }
  }
]

// 找出未使用的索引
const indexes = db.collection.getIndexes();
const stats = db.collection.aggregate([{ $indexStats: {} }]).toArray();
const usedIndexes = new Set(stats.map(s => s.name));
indexes.forEach(idx => {
  if (!usedIndexes.has(idx.name)) {
    print(`Unused index: ${idx.name}`);
  }
});
```

## Profiler

### 启用 Profiler

```javascript
// 启用 Profiler
// level: 0=关闭, 1=慢查询, 2=所有查询
db.setProfilingLevel(1, { slowms: 100 })  // 记录超过 100ms 的查询

// 查看当前配置
db.getProfilingStatus()
// { was: 1, slowms: 100, sampleRate: 1.0 }

// 关闭 Profiler
db.setProfilingLevel(0)
```

### 查看慢查询

```javascript
// 查看最近的慢查询
db.system.profile.find().sort({ ts: -1 }).limit(10)

// 查看特定集合的慢查询
db.system.profile.find(
  { ns: "shop.orders" }
).sort({ millis: -1 }).limit(10)

// 格式化输出
db.system.profile.find({ millis: { $gt: 100 } }).forEach(doc => {
  print(`\n=== Query ===`);
  print(`Time: ${doc.ts}`);
  print(`Duration: ${doc.millis}ms`);
  print(`Command: ${JSON.stringify(doc.command)}`);
  print(`Docs scanned: ${doc.nscanned}`);
  print(`Docs returned: ${doc.nreturned}`);
})
```

### Profiler 字段说明

```javascript
// Profiler 记录字段
{
  op: "query",              // 操作类型
  ns: "shop.orders",        // 命名空间
  command: { ... },          // 完整命令
  docsExamined: 10000,       // 检查的文档数
  nreturned: 10,             // 返回的文档数
  millis: 150,               // 执行时间（毫秒）
  ts: ISODate("..."),        // 时间戳
  client: "192.168.1.100",   // 客户端地址
  user: "admin"              // 用户
}
```

## 慢查询分析

### 分析慢查询

```javascript
// 1. 找出最慢的查询
db.system.profile.find().sort({ millis: -1 }).limit(5).forEach(doc => {
  print(`${doc.millis}ms - ${doc.ns}`);
});

// 2. 分析扫描文档数 vs 返回数
// docsExamined 远大于 nreturned 说明需要优化
db.system.profile.find().forEach(doc => {
  const ratio = doc.docsExamined / (doc.nreturned || 1);
  if (ratio > 100) {
    print(`High scan ratio (${ratio}): ${doc.ns}`);
  }
});

// 3. 检查是否使用索引
db.system.profile.find({ docsExamined: { $gt: 10000 } }).forEach(doc => {
  print(`Full scan: ${doc.ns}, docs: ${doc.docsExamined}`);
});
```

### 优化慢查询

```javascript
// 1. 检查索引
db.collection.find({ field: value }).explain()

// 2. 添加索引
db.collection.createIndex({ field: 1 })

// 3. 重写查询
// ❌ 不好
db.orders.find({ status: "completed" }).limit(100)

// ✅ 好（使用索引）
db.orders.find({ status: "completed", _id: { $gt: lastId } }).limit(100)
```

## 监控脚本

### 综合监控脚本

```javascript
// MongoDB 监控脚本
function mongoMonitor() {
  const status = db.serverStatus();

  print("=== MongoDB Monitor ===");
  print(`Time: ${new Date()}`);
  print(`Uptime: ${Math.floor(status.uptime / 86400)}d ${Math.floor((status.uptime % 86400) / 3600)}h`);

  // 连接
  const conn = status.connections;
  print(`\nConnections: ${conn.current} (max: ${conn.available})`);

  // 操作
  const ops = status.opcounters;
  print(`\nOperations:`);
  print(`  Insert: ${ops.insert}`);
  print(`  Query: ${ops.query}`);
  print(`  Update: ${ops.update}`);
  print(`  Delete: ${ops.delete}`);

  // 内存
  const cache = status.wiredTiger.cache;
  print(`\nCache:`);
  print(`  Usage: ${cache['percentage of cache in use']}%`);
  print(`  Dirty: ${(cache['cache dirty bytes'] / 1024 / 1024).toFixed(2)} MB`);

  // 复制
  const repl = status.repl;
  if (repl.primary) {
    print(`\nReplication:`);
    print(`  Primary: ${repl.primary}`);
    print(`  Secondary: ${repl.secondary}`);
  }
}

// 运行
mongoMonitor();
```

## 告警配置

### 关键告警指标

```javascript
// 推荐告警阈值：

// 1. 连接数 > 80%
db.serverStatus().connections.current > db.serverStatus().connections.available * 0.8

// 2. Cache 使用率 > 90%
db.serverStatus().wiredTiger.cache['percentage of cache in use'] > 90

// 3. 复制延迟 > 30 秒
db.serverStatus().repl.secondaryDelaySecs > 30

// 4. 队列长度 > 10
db.serverStatus().connections.current > 10

// 5. 慢查询数量突增
db.system.profile.countDocuments({ millis: { $gt: 1000 }, ts: { $gt: new Date(Date.now() - 60000) } }) > 10
```

## 总结

监控要点：

- **serverStatus**：综合监控所有指标
- **mongostat**：实时监控操作和资源
- **mongotop**：分析集合读写时间
- **Profiler**：追踪慢查询
- **stats**：分析集合和索引使用

定期监控是保证 MongoDB 稳定运行的关键。
