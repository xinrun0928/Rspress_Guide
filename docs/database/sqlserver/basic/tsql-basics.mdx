# T-SQL 基础

对于熟悉 MySQL 的开发者，SQL Server 的 T-SQL 语法并不陌生。两者都遵循 ANSI SQL 标准，但各有扩展和差异。

本文聚焦 SQL Server 特有的语法和最佳实践。

## 数据类型

### 常用数据类型

| 类型 | 范围/大小 | 说明 |
|---|---|---|
| `INT` | -21亿~21亿 | 32 位整数 |
| `BIGINT` | ±9万亿 | 64 位整数 |
| `DECIMAL(p,s)` | 可变 | 定点数，p=精度，s=小数位 |
| `FLOAT` | 可变 | 浮点数 |
| `DATETIME2` | 0001-01-01~9999 | 时间戳，精度更高 |
| `DATE` | 同上 | 仅日期 |
| `TIME` | 同上 | 仅时间 |
| `CHAR(n)` | n 字符 | 固定长度 |
| `VARCHAR(n)` | n 字符 | 可变长度 |
| `VARCHAR(MAX)` | 2GB | 大文本 |
| `NVARCHAR(n/MAX)` | Unicode | Unicode 字符 |
| `UNIQUEIDENTIFIER` | 16 字节 | GUID/UUID |
| `ROWVERSION` | 8 字节 | 自动递增二进制 |

```sql
-- 日期时间推荐使用 DATETIME2(3)
CREATE TABLE Orders (
    OrderID INT IDENTITY(1,1) PRIMARY KEY,
    OrderDate DATETIME2(3) NOT NULL DEFAULT SYSDATETIME(),
    CustomerName NVARCHAR(100),
    Amount DECIMAL(10,2),
    RowVer ROWVERSION  -- 自动版本号
);
```

:::warning
避免使用 `DATETIME`（精度 3.33ms），改用 `DATETIME2(7)`（精度 100ns）。`DATETIME` 与 `DATE` 混合计算时可能有边界问题。
:::

## DDL 操作

### 创建表

```sql
CREATE TABLE Products (
    ProductID INT IDENTITY(1,1) PRIMARY KEY,
    ProductName NVARCHAR(100) NOT NULL,
    CategoryID INT,
    Price DECIMAL(10,2) CHECK (Price >= 0),
    CreatedDate DATETIME2 DEFAULT SYSDATETIME(),
    ModifiedDate DATETIME2,
    CONSTRAINT FK_Products_Categories FOREIGN KEY (CategoryID) 
        REFERENCES Categories(CategoryID)
);

-- 使用 schema
CREATE TABLE dbo.Products (...);
```

### 修改表

```sql
-- 添加列
ALTER TABLE Products ADD Description NVARCHAR(500);

-- 删除列
ALTER TABLE Products DROP COLUMN Description;

-- 添加约束
ALTER TABLE Products ADD CONSTRAINT DF_Products_CreatedDate 
    DEFAULT SYSDATETIME() FOR CreatedDate;

-- 修改列类型（需要重新计算所有数据）
ALTER TABLE Products ALTER COLUMN Price DECIMAL(12,2);
```

### 删除表

```sql
-- 删除表
DROP TABLE Products;

-- 如果存在才删除
DROP TABLE IF EXISTS Products;  -- SQL Server 2016+

-- 截断表（更快，不记录日志，但不可回滚）
TRUNCATE TABLE Products;
```

## DML 操作

### 插入数据

```sql
-- 单行插入
INSERT INTO Products (ProductName, CategoryID, Price)
VALUES ('Widget', 1, 19.99);

-- 多行插入
INSERT INTO Products (ProductName, CategoryID, Price)
VALUES 
    ('Widget A', 1, 19.99),
    ('Widget B', 1, 24.99),
    ('Gadget', 2, 49.99);

-- 从其他表插入
INSERT INTO Products (ProductName, CategoryID, Price)
SELECT ProductName, CategoryID, Price
FROM StagingProducts
WHERE IsActive = 1;

-- OUTPUT 子句
INSERT INTO Products (ProductName, CategoryID, Price)
OUTPUT inserted.ProductID, inserted.ProductName
VALUES ('Widget', 1, 19.99);
```

### 更新数据

```sql
-- 普通更新
UPDATE Products
SET Price = Price * 1.1,
    ModifiedDate = SYSDATETIME()
WHERE CategoryID = 1;

-- 使用 CTE 更新
WITH CategoryProducts AS (
    SELECT ProductID, Price, Price * 1.1 AS NewPrice
    FROM Products
    WHERE CategoryID = 1
)
UPDATE CategoryProducts
SET Price = NewPrice;

-- 使用 OUTPUT
UPDATE Products
SET Price = Price * 1.1
OUTPUT deleted.ProductID, deleted.Price AS OldPrice, inserted.Price AS NewPrice
WHERE CategoryID = 1;
```

### 删除数据

```sql
-- 普通删除
DELETE FROM Products WHERE ProductID = 10;

-- 使用 TOP 限制
DELETE TOP (1000) FROM Products
WHERE IsArchived = 1;

-- OUTPUT 子句
DELETE FROM Products
OUTPUT deleted.ProductID, deleted.ProductName
WHERE ProductID = 10;

-- 使用 MERGE
MERGE INTO TargetTable AS T
USING SourceTable AS S ON T.ID = S.ID
WHEN MATCHED THEN
    UPDATE SET T.Value = S.Value
WHEN NOT MATCHED THEN
    INSERT (ID, Value) VALUES (S.ID, S.Value)
WHEN NOT MATCHED BY SOURCE THEN
    DELETE;
```

## 查询

### 基本查询

```sql
-- WHERE 条件
SELECT ProductID, ProductName, Price
FROM Products
WHERE Price BETWEEN 10 AND 100
    AND CategoryID IN (1, 2, 3)
    AND ProductName LIKE 'W%'
ORDER BY Price DESC, ProductName ASC
OFFSET 0 ROWS FETCH NEXT 10 ROWS ONLY;

-- DISTINCT
SELECT DISTINCT CategoryID FROM Products;

-- TOP N
SELECT TOP 10 * FROM Products ORDER BY Price DESC;
```

### 聚合查询

```sql
-- 聚合函数
SELECT 
    COUNT(*) AS TotalCount,
    COUNT(DISTINCT CategoryID) AS CategoryCount,
    SUM(Price) AS TotalPrice,
    AVG(Price) AS AvgPrice,
    MIN(Price) AS MinPrice,
    MAX(Price) AS MaxPrice
FROM Products
WHERE IsActive = 1;

-- GROUP BY + HAVING
SELECT 
    CategoryID,
    COUNT(*) AS ProductCount,
    AVG(Price) AS AvgPrice
FROM Products
GROUP BY CategoryID
HAVING COUNT(*) > 5;
```

### 联接

```sql
-- INNER JOIN
SELECT o.OrderID, c.CustomerName, o.Amount
FROM Orders o
INNER JOIN Customers c ON o.CustomerID = c.CustomerID;

-- LEFT JOIN
SELECT c.CustomerName, o.OrderID
FROM Customers c
LEFT JOIN Orders o ON c.CustomerID = o.CustomerID
WHERE o.OrderID IS NULL;  -- 没有订单的客户

-- 多表联接
SELECT o.OrderID, c.CustomerName, p.ProductName, od.Quantity
FROM Orders o
JOIN Customers c ON o.CustomerID = c.CustomerID
JOIN OrderDetails od ON o.OrderID = od.OrderID
JOIN Products p ON od.ProductID = p.ProductID;
```

### 子查询

```sql
-- 标量子查询
SELECT ProductName, Price,
    (SELECT AVG(Price) FROM Products) AS AvgPrice
FROM Products;

-- 表子查询
SELECT * FROM Products
WHERE CategoryID IN (
    SELECT CategoryID FROM Categories WHERE IsActive = 1
);

-- EXISTS
SELECT c.CustomerName
FROM Customers c
WHERE EXISTS (
    SELECT 1 FROM Orders o WHERE o.CustomerID = c.CustomerID
);
```

### 公用表表达式（CTE）

```sql
-- 基本 CTE
WITH ActiveProducts AS (
    SELECT ProductID, ProductName, Price, CategoryID
    FROM Products
    WHERE IsActive = 1
)
SELECT * FROM ActiveProducts WHERE Price > 50;

-- 递归 CTE
WITH CategoryTree AS (
    -- 基础查询：顶级分类
    SELECT CategoryID, CategoryName, ParentCategoryID, 0 AS Level
    FROM Categories
    WHERE ParentCategoryID IS NULL
    
    UNION ALL
    
    -- 递归查询：子分类
    SELECT c.CategoryID, c.CategoryName, c.ParentCategoryID, ct.Level + 1
    FROM Categories c
    JOIN CategoryTree ct ON c.ParentCategoryID = ct.CategoryID
)
SELECT * FROM CategoryTree ORDER BY Level, CategoryName;
```

## 控制流

```sql
-- IF...ELSE
IF (SELECT COUNT(*) FROM Products WHERE Price < 0) > 0
BEGIN
    PRINT 'Warning: Negative prices found';
    UPDATE Products SET Price = 0 WHERE Price < 0;
END
ELSE
    PRINT 'All prices are valid';

-- CASE 表达式
SELECT 
    ProductName,
    Price,
    CASE 
        WHEN Price < 10 THEN 'Low'
        WHEN Price < 50 THEN 'Medium'
        WHEN Price < 100 THEN 'High'
        ELSE 'Premium'
    END AS PriceCategory
FROM Products;

-- WHILE 循环
DECLARE @Counter INT = 1;
WHILE @Counter <= 10
BEGIN
    PRINT 'Iteration ' + CAST(@Counter AS VARCHAR);
    SET @Counter = @Counter + 1;
END;

-- TRY...CATCH
BEGIN TRY
    INSERT INTO Products (ProductName, Price) VALUES ('Test', -1);
END TRY
BEGIN CATCH
    PRINT 'Error: ' + ERROR_MESSAGE();
    PRINT 'Line: ' + CAST(ERROR_LINE() AS VARCHAR);
END CATCH;
```

## 变量

```sql
-- 声明变量
DECLARE @ProductID INT;
DECLARE @ProductName NVARCHAR(100) = 'Default';
DECLARE @StartDate DATETIME2 = '2024-01-01';

-- SET vs SELECT
SET @ProductID = 1;  -- 标量值
SELECT @ProductID = ProductID FROM Products WHERE ProductName = 'Widget';
SELECT @ProductID = ProductID, @ProductName = ProductName FROM Products;

-- 表变量
DECLARE @Products TABLE (
    ProductID INT PRIMARY KEY,
    ProductName NVARCHAR(100),
    Price DECIMAL(10,2)
);

INSERT INTO @Products SELECT * FROM Products WHERE Price > 100;
SELECT * FROM @Products;
```

:::tip
表变量（`@table`）适合小数据量（`< 1000 行`），临时表（`#table`）适合大数据量。表变量没有统计信息，可能导致糟糕的执行计划。
:::

## 函数

### 内置函数

```sql
-- 字符串函数
SELECT 
    '  Hello  ' AS Original,
    LTRIM('  Hello  ') AS LTrim,
    RTRIM('  Hello  ') AS RTrim,
    LEN('Hello') AS Length,           -- 不计尾部空格
    DATALENGTH('Hello') AS DataLength, -- 计字节
    UPPER('hello') AS Upper,
    LOWER('HELLO') AS Lower,
    SUBSTRING('Hello World', 1, 5) AS Sub,
    CHARINDEX('World', 'Hello World') AS IndexOf,
    CONCAT('Hello', ' ', 'World') AS Concat;

-- 日期函数
SELECT 
    GETDATE() AS CurrentDate,
    GETUTCDATE() AS UTCDate,
    SYSDATETIME() AS SysDateTime,
    DATEADD(DAY, 7, GETDATE()) AS NextWeek,
    DATEDIFF(DAY, '2024-01-01', GETDATE()) AS DaysSince,
    DATEFROMPARTS(2024, 3, 15) AS Date,
    EOMONTH(GETDATE()) AS EndOfMonth;

-- 转换函数
SELECT 
    CAST(123.45 AS VARCHAR(10)) AS CastResult,
    CONVERT(VARCHAR(10), GETDATE(), 120) AS ConvertResult,
    TRY_CAST('abc' AS INT) AS TryCastResult,  -- SQL Server 2012+
    FORMAT(GETDATE(), 'yyyy-MM-dd') AS Formatted;
```

### 自定义函数

```sql
-- 标量函数
CREATE FUNCTION dbo.fn_CalculateDiscount
(
    @Price DECIMAL(10,2),
    @DiscountPercent INT
)
RETURNS DECIMAL(10,2)
AS
BEGIN
    RETURN @Price * (1 - @DiscountPercent / 100.0);
END;

-- 表值函数
CREATE FUNCTION dbo.fn_GetProductsByCategory
(
    @CategoryID INT
)
RETURNS TABLE
AS
RETURN
(
    SELECT ProductID, ProductName, Price
    FROM Products
    WHERE CategoryID = @CategoryID
);

-- 使用表值函数
SELECT * FROM dbo.fn_GetProductsByCategory(1);
```

## 批处理

```sql
-- 批处理分隔符
SELECT * FROM Products;
GO

UPDATE Products SET ModifiedDate = GETDATE();
GO

-- 动态 SQL
DECLARE @TableName NVARCHAR(100) = 'Products';
DECLARE @SQL NVARCHAR(MAX);

SET @SQL = N'SELECT COUNT(*) FROM ' + QUOTENAME(@TableName);
EXEC sp_executesql @SQL;

-- 参数化动态 SQL（防止 SQL 注入）
SET @SQL = N'SELECT * FROM Products WHERE CategoryID = @CategoryID';
EXEC sp_executesql @SQL, N'@CategoryID INT', @CategoryID = 1;
```

## 常用系统表和 DMV

```sql
-- 查看表结构
EXEC sp_help 'Products';

-- 查看索引
EXEC sp_helpindex 'Products';

-- 查看约束
SELECT 
    OBJECT_NAME(parent_object_id) AS TableName,
    name AS ConstraintName,
    type_desc AS ConstraintType
FROM sys.key_constraints
WHERE type = 'PK';

-- 查看依赖关系
SELECT 
    referencing_schema_name + '.' + referencing_entity_name AS ReferencingObject,
    referenced_schema_name + '.' + referenced_entity_name AS ReferencedObject
FROM sys.sql_expression_dependencies
WHERE referencing_id = OBJECT_ID('YourFunction');
```
