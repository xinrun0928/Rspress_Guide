# Oracle 常用函数

Oracle 内置了数百个函数，覆盖字符串处理、日期运算、数值计算、数据类型转换、聚合分析等场景。熟练使用这些函数，可以让 SQL 更加简洁、高效。

但面试官真正想看的不是你会不会"查文档"，而是你是否理解这些函数背后的原理，是否知道在什么场景下选择什么函数，以及如何避免常见的性能陷阱。

## 字符函数

### 大小写转换

```sql
-- LOWER：转小写
SELECT LOWER('Hello World') FROM DUAL;  -- hello world

-- UPPER：转大写
SELECT UPPER('Hello World') FROM DUAL;  -- HELLO WORLD

-- INITCAP：首字母大写
SELECT INITCAP('hello world') FROM DUAL;  -- Hello World
```

### 字符串操作

```sql
-- LENGTH：返回字符数
SELECT LENGTH('你好世界') FROM DUAL;  -- 4

-- LENGTHB：返回字节数（中文在 UTF-8 下每个汉字 3 字节）
SELECT LENGTHB('你好世界') FROM DUAL;  -- 12

-- SUBSTR：截取字符串（从 1 开始）
SELECT SUBSTR('Hello World', 1, 5) FROM DUAL;   -- Hello
SELECT SUBSTR('Hello World', -5) FROM DUAL;     -- World

-- SUBSTRB：按字节截取
SELECT SUBSTRB('你好世界', 1, 4) FROM DUAL;  -- 你好

-- INSTR：查找子串位置（从 1 开始，未找到返回 0）
SELECT INSTR('Hello World', 'or') FROM DUAL;    -- 8
SELECT INSTR('Hello World', 'x') FROM DUAL;      -- 0

-- TRIM、LTRIM、RTRIM：去除空格
SELECT TRIM('  Hello  ') FROM DUAL;        -- Hello
SELECT LTRIM('  Hello  ') FROM DUAL;       -- Hello  
SELECT RTRIM('  Hello  ') FROM DUAL;       --   Hello

-- TRIM 去除指定字符
SELECT TRIM(LEADING '0' FROM '00012300') FROM DUAL;  -- 12300
SELECT TRIM(TRAILING '0' FROM '00012300') FROM DUAL; -- 000123
SELECT TRIM(BOTH '0' FROM '00012300') FROM DUAL;     -- 123

-- LPAD、RPAD：补齐长度
SELECT LPAD('123', 8, '0') FROM DUAL;  -- 00000123
SELECT RPAD('Hello', 10, '*') FROM DUAL;  -- Hello*****
```

### 字符串拼接

```sql
-- || 操作符（推荐）
SELECT 'Hello' || ' ' || 'World' FROM DUAL;  -- Hello World

-- CONCAT 函数
SELECT CONCAT('Hello', 'World') FROM DUAL;  -- HelloWorld

-- REGEXP_REPLACE：正则替换
SELECT REGEXP_REPLACE('abc123def456', '[0-9]+', '#') FROM DUAL;  -- abc#def#

-- REGEXP_INSTR：正则查找位置
SELECT REGEXP_INSTR('abc123def', '[0-9]+') FROM DUAL;  -- 4
```

## 数值函数

```sql
-- ROUND：四舍五入
SELECT ROUND(3.14159, 2) FROM DUAL;   -- 3.14
SELECT ROUND(3.14159, 0) FROM DUAL;   -- 3
SELECT ROUND(3159, -2) FROM DUAL;      -- 3200

-- TRUNC：截断
SELECT TRUNC(3.14159, 2) FROM DUAL;   -- 3.14
SELECT TRUNC(3.99999, 0) FROM DUAL;   -- 3

-- MOD：取余
SELECT MOD(10, 3) FROM DUAL;           -- 1

-- POWER：幂运算
SELECT POWER(2, 10) FROM DUAL;        -- 1024

-- SQRT：平方根
SELECT SQRT(16) FROM DUAL;            -- 4

-- ABS：绝对值
SELECT ABS(-10) FROM DUAL;            -- 10

-- CEIL、FLOOR：向上/下取整
SELECT CEIL(3.1) FROM DUAL;            -- 4
SELECT FLOOR(3.9) FROM DUAL;           -- 3

-- SIGN：符号函数
SELECT SIGN(-10) FROM DUAL;           -- -1
SELECT SIGN(0) FROM DUAL;             -- 0
SELECT SIGN(10) FROM DUAL;             -- 1
```

## 日期函数

```sql
-- SYSDATE vs SYSTIMESTAMP
SELECT SYSDATE FROM DUAL;        -- 2024-03-15 12:00:00
SELECT SYSTIMESTAMP FROM DUAL;  -- 2024-03-15 12:00:00.123456 +08:00

-- 日期加减（加数字表示加天数）
SELECT SYSDATE + 1 AS tomorrow,
       SYSDATE - 1 AS yesterday
FROM DUAL;

-- ADD_MONTHS：加减月份
SELECT ADD_MONTHS(SYSDATE, 6) FROM DUAL;
SELECT ADD_MONTHS(SYSDATE, -12) FROM DUAL;  -- 一年前

-- MONTHS_BETWEEN：月份差
SELECT MONTHS_BETWEEN(SYSDATE, TO_DATE('2023-01-01', 'YYYY-MM-DD')) 
FROM DUAL;

-- LAST_DAY：月份最后一天
SELECT LAST_DAY(SYSDATE) FROM DUAL;

-- NEXT_DAY：下周指定星期几
SELECT NEXT_DAY(SYSDATE, '星期一') FROM DUAL;
SELECT NEXT_DAY(SYSDATE, 2) FROM DUAL;  -- 2 表示星期一，7 表示星期日

-- EXTRACT：提取日期分量
SELECT EXTRACT(YEAR FROM SYSDATE) AS year,
       EXTRACT(MONTH FROM SYSDATE) AS month,
       EXTRACT(DAY FROM SYSDATE) AS day,
       EXTRACT(HOUR FROM SYSTIMESTAMP) AS hour,
       EXTRACT(MINUTE FROM SYSTIMESTAMP) AS minute
FROM DUAL;
```

## 转换函数

### TO_CHAR：转换为字符串

```sql
-- 日期转字符串
SELECT TO_CHAR(SYSDATE, 'YYYY-MM-DD') FROM DUAL;        -- 2024-03-15
SELECT TO_CHAR(SYSDATE, 'YYYY"年"MM"月"DD"日"') FROM DUAL;  -- 2024年03月15日
SELECT TO_CHAR(SYSDATE, 'HH24:MI:SS') FROM DUAL;        -- 14:30:00
SELECT TO_CHAR(SYSDATE, 'DAY') FROM DUAL;               -- 星期五
SELECT TO_CHAR(SYSDATE, 'DY') FROM DUAL;                -- 星期五

-- 数值转字符串
SELECT TO_CHAR(1234.567, '999,999.99') FROM DUAL;  --  1,234.57
SELECT TO_CHAR(1234.567, '000,000.00') FROM DUAL;  -- 001,234.57
SELECT TO_CHAR(1234.5, '$999,999.00') FROM DUAL;   --  $1,234.50
```

### TO_NUMBER：转换为数值

```sql
SELECT TO_NUMBER('12345') + 1 FROM DUAL;  -- 12346
SELECT TO_NUMBER('$1,234.56', '9999.99') FROM DUAL;  -- 需配合格式模型
```

### TO_DATE：转换为日期

```sql
SELECT TO_DATE('2024-03-15', 'YYYY-MM-DD') FROM DUAL;
SELECT TO_DATE('202403151430', 'YYYYMMDDHH24MI') FROM DUAL;
```

### CAST 与 CONVERT

```sql
-- CAST：通用类型转换
SELECT CAST('123' AS NUMBER) FROM DUAL;
SELECT CAST(SYSDATE AS VARCHAR2(20)) FROM DUAL;
SELECT CAST(123.456 AS VARCHAR2(10)) FROM DUAL;

-- CONVERT：字符集转换
SELECT CONVERT('Hello', 'US7ASCII', 'WE8ISO8859P1') FROM DUAL;
```

## 聚合函数

```sql
-- COUNT：计数
SELECT COUNT(*) FROM employees;               -- 所有行
SELECT COUNT(DISTINCT department_id) FROM employees;  -- 去重计数
SELECT COUNT(commission_pct) FROM employees;  -- 只统计非 NULL

-- SUM：求和
SELECT SUM(salary) FROM employees;

-- AVG：平均值（忽略 NULL）
SELECT AVG(NVL(commission_pct, 0)) FROM employees;

-- MAX、MIN：最大最小值
SELECT MAX(salary), MIN(salary) FROM employees;

-- LISTAGG：字符串聚合（Oracle 11g R2+）
SELECT department_id,
       LISTAGG(last_name, ',') WITHIN GROUP (ORDER BY hire_date) AS members
FROM employees
GROUP BY department_id;
```

### KEEP 关键字

```sql
-- KEEP (DENSE_RANK FIRST/LAST ...)：保留第一条或最后一条
SELECT department_id,
       MAX(salary) KEEP (DENSE_RANK FIRST ORDER BY hire_date) AS earliest_hire_salary
FROM employees
GROUP BY department_id;
```

## 分析函数

分析函数是 Oracle SQL 最强大的特性之一，用于在窗口内进行计算。

```sql
-- ROW_NUMBER、RANK、DENSE_RANK：排名
SELECT employee_id, salary,
       ROW_NUMBER() OVER (ORDER BY salary DESC) AS row_num,
       RANK() OVER (ORDER BY salary DESC) AS rank,
       DENSE_RANK() OVER (ORDER BY salary DESC) AS dense_rank
FROM employees;

-- LAG、LEAD：获取前一行/后一行
SELECT employee_id, hire_date, salary,
       LAG(salary, 1) OVER (ORDER BY hire_date) AS prev_salary,
       LEAD(salary, 1) OVER (ORDER BY hire_date) AS next_salary
FROM employees;

-- SUM、AVG 窗口函数
SELECT employee_id, hire_date, salary,
       SUM(salary) OVER (ORDER BY hire_date) AS running_total,
       AVG(salary) OVER (ORDER BY hire_date 
                         ROWS BETWEEN 2 PRECEDING AND CURRENT ROW) AS moving_avg
FROM employees;

-- FIRST_VALUE、LAST_VALUE
SELECT department_id, employee_id, salary,
       FIRST_VALUE(salary) OVER (PARTITION BY department_id ORDER BY hire_date) AS first_salary,
       LAST_VALUE(salary) OVER (PARTITION BY department_id ORDER BY hire_date 
                                 ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) AS last_salary
FROM employees;
```

### 窗口子句详解

```sql
-- 窗口子句语法
SELECT ... OVER (
    [PARTITION BY col_list]
    [ORDER BY order_list]
    [ROWS/RANGE window_frame]
)

-- 窗口框架示例
ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW      -- 从开始到当前行
ROWS BETWEEN CURRENT ROW AND UNBOUNDED FOLLOWING      -- 从当前行到结束
ROWS BETWEEN 2 PRECEDING AND 2 FOLLOWING             -- 当前行前后各 2 行
RANGE BETWEEN INTERVAL '7' DAY PRECEDING AND CURRENT ROW  -- 按值窗口
```

## 常见面试问题

**问：NVL 和 COALESCE 的区别是什么？**

NVL 只接受两个参数，当第一个参数为 NULL 时返回第二个参数。COALESCE 接受多个参数，返回第一个非 NULL 值。NVL 等价于 `COALESCE(expr1, expr2)`，但 COALESCE 更通用，可以处理多个表达式。

**问：KEEP (DENSE_RANK FIRST ...) 的作用是什么？**

KEEP 子句与聚合函数配合使用，用于在分组后保留排序后的第一条或最后一条记录。例如 `MAX(salary) KEEP (DENSE_RANK FIRST ORDER BY hire_date)` 返回每个部门最早入职员工的工资。

**问：分析函数和聚合函数的区别是什么？**

聚合函数将多行数据聚合成一行（GROUP BY）；分析函数在保持原有多行输出的同时，在每一行上计算基于窗口的结果。分析函数不会减少行数，常用于计算排名、累计、移动平均等。

## 总结

Oracle 函数丰富多样，涵盖字符串、数值、日期、转换、聚合、分析等各类场景。掌握这些函数的用法，特别是分析函数，可以让你的 SQL 更加简洁高效，同时展示 Oracle 技术深度。
