# Oracle 集合类型：VARRAY、嵌套表、关联数组

在 PL/SQL 中处理多行数据时，除了使用游标逐行处理，你还可以使用集合类型将数据一次性加载到内存中进行操作。集合就像 PL/SQL 中的"数组"，支持批量操作、高效查找和灵活的数据结构组织。

假设你需要从数据库中获取一个部门的员工列表，然后在 PL/SQL 中进行过滤、排序、聚合等操作。使用集合类型，这些操作可以在内存中快速完成，无需多次数据库往返。

## 集合类型概述

Oracle PL/SQL 提供三种集合类型：

| 类型 | 说明 | 下标范围 | 存储位置 | 特点 |
|---|---|---|---|---|
| 关联数组（Associative Array） | 键值对 | 字符串或整数 | PGA（内存） | 稀疏，最灵活 |
| 嵌套表（Nested Table） | 无序集合 | 整数（1~2GB） | PGA 或数据库 | 可持久化到表 |
| VARRAY | 可变数组 | 整数（固定上限） | PGA 或数据库 | 有边界，大小固定 |

## 关联数组（Associative Array）

关联数组是最灵活的集合类型，下标可以是整数或字符串，非常适合实现键值查找。

### 整数下标

```sql
DECLARE
    TYPE emp_salary_table IS TABLE OF NUMBER INDEX BY PLS_INTEGER;
    v_salaries emp_salary_table;
BEGIN
    -- 赋值
    v_salaries(100) := 5000;
    v_salaries(101) := 6000;
    v_salaries(200) := 7500;
    
    -- 访问
    DBMS_OUTPUT.PUT_LINE('员工 100 工资: ' || v_salaries(100));
    
    -- 检查是否存在
    IF v_salaries.EXISTS(300) THEN
        DBMS_OUTPUT.PUT_LINE('存在');
    ELSE
        DBMS_OUTPUT.PUT_LINE('员工 300 不存在');
    END IF;
    
    -- 遍历
    FOR i IN v_salaries.FIRST..v_salaries.LAST LOOP
        DBMS_OUTPUT.PUT_LINE('Key: ' || i || ', Value: ' || v_salaries(i));
    END LOOP;
END;
/
```

### 字符串下标

```sql
DECLARE
    TYPE emp_info_table IS TABLE OF VARCHAR2(100) INDEX BY VARCHAR2(50);
    v_emp_info emp_info_table;
BEGIN
    v_emp_info('Zhang San') := 'Engineer';
    v_emp_info('Li Si') := 'Manager';
    v_emp_info('Wang Wu') := 'Analyst';
    
    DBMS_OUTPUT.PUT_LINE(v_emp_info('Zhang San'));
    
    -- 遍历
    DECLARE
        v_key VARCHAR2(50);
    BEGIN
        v_key := v_emp_info.FIRST;
        WHILE v_key IS NOT NULL LOOP
            DBMS_OUTPUT.PUT_LINE(v_key || ': ' || v_emp_info(v_key));
            v_key := v_emp_info.NEXT(v_key);
        END LOOP;
    END;
END;
/
```

### 关联数组用于批量查询

```sql
DECLARE
    TYPE emp_record IS RECORD (
        emp_id    employees.employee_id%TYPE,
        emp_name  VARCHAR2(100),
        salary    employees.salary%TYPE
    );
    TYPE emp_table IS TABLE OF emp_record INDEX BY PLS_INTEGER;
    
    v_emps emp_table;
BEGIN
    SELECT employee_id, first_name || ' ' || last_name, salary
    BULK COLLECT INTO v_emps
    FROM employees
    WHERE department_id = 50;
    
    FOR i IN 1..v_emps.COUNT LOOP
        DBMS_OUTPUT.PUT_LINE(v_emps(i).emp_name || ': ' || v_emps(i).salary);
    END LOOP;
END;
/
```

## 嵌套表（Nested Table）

嵌套表是 Oracle 数据库支持的可持久化集合类型，可以作为表列的数据类型存储在数据库中。

### 定义嵌套表类型

```sql
-- 在包中定义类型
CREATE OR REPLACE PACKAGE emp_pkg AS
    TYPE emp_name_list IS TABLE OF VARCHAR2(100);
END emp_pkg;
/

-- 作为表列类型
CREATE TYPE phone_list_type AS TABLE OF VARCHAR2(20);
/

CREATE TABLE contacts (
    contact_id NUMBER PRIMARY KEY,
    contact_name VARCHAR2(100),
    phone_numbers phone_list_type  -- 嵌套表列
) NESTED TABLE phone_numbers STORE AS phone_numbers_tab;
/

-- 插入数据
INSERT INTO contacts VALUES (
    1, '张三',
    phone_list_type('13800138000', '13900139000')
);

INSERT INTO contacts VALUES (
    2, '李四',
    phone_list_type('13700137000')
);
/

-- 查询嵌套表
SELECT contact_name, t.* FROM contacts c, TABLE(c.phone_numbers) t
WHERE c.contact_id = 1;

-- 更新嵌套表
UPDATE TABLE (
    SELECT phone_numbers FROM contacts WHERE contact_id = 1
) SET COLUMN_VALUE = '13888888888' WHERE COLUMN_VALUE = '13800138000';
/

-- 添加元素
INSERT INTO TABLE (
    SELECT phone_numbers FROM contacts WHERE contact_id = 1
) VALUES ('13600136000');
/

-- 删除元素
DELETE FROM TABLE (
    SELECT phone_numbers FROM contacts WHERE contact_id = 1
) WHERE COLUMN_VALUE = '13900139000';
/
```

### PL/SQL 中的嵌套表

```sql
DECLARE
    TYPE int_list IS TABLE OF NUMBER;
    v_numbers int_list := int_list(1, 2, 3, 4, 5);
BEGIN
    -- 使用构造函数初始化
    v_numbers := int_list(10, 20, 30);
    
    -- 添加元素
    v_numbers.EXTEND;           -- 扩展一个空元素
    v_numbers(v_numbers.COUNT) := 40;
    
    v_numbers.EXTEND(3);        -- 扩展三个空元素
    v_numbers(v_numbers.COUNT-2) := 50;
    v_numbers(v_numbers.COUNT-1) := 60;
    v_numbers(v_numbers.COUNT) := 70;
    
    -- 删除元素（保留空间）
    v_numbers.DELETE(2);
    
    -- 遍历
    FOR i IN 1..v_numbers.COUNT LOOP
        IF v_numbers.EXISTS(i) THEN  -- 检查是否存在
            DBMS_OUTPUT.PUT_LINE(v_numbers(i));
        END IF;
    END LOOP;
END;
/
```

## VARRAY（可变数组）

VARRAY 是有固定上限的集合类型，适合"元素数量有明确上限"的场景。

```sql
DECLARE
    -- 定义 VARRAY，最多 10 个元素
    TYPE week_days IS VARRAY(7) OF VARCHAR2(20);
    v_days week_days := week_days('Mon', 'Tue', 'Wed');
BEGIN
    -- 添加元素（不能超过 10 个）
    v_days.EXTEND;
    v_days(4) := 'Thu';
    
    v_days.EXTEND(3);
    v_days(5) := 'Fri';
    v_days(6) := 'Sat';
    v_days(7) := 'Sun';
    
    -- 尝试超过上限会报错
    -- v_days.EXTEND;  -- ORA-06532: Subscript outside of limit
    
    -- 遍历
    FOR i IN 1..v_days.COUNT LOOP
        DBMS_OUTPUT.PUT_LINE(v_days(i));
    END LOOP;
END;
/
```

## 集合方法（Methods）

所有集合类型都支持以下方法：

```sql
DECLARE
    TYPE num_list IS TABLE OF NUMBER;
    v_nums num_list := num_list(10, 20, 30, NULL, 40, 50);
BEGIN
    v_nums.COUNT;         -- 元素数量（3，不含已删除的）
    v_nums.LIMIT;         -- 最大元素数（VARRAY 有上限，嵌套表/关联数组 无上限返回 NULL）
    v_nums.FIRST;         -- 第一个下标
    v_nums.LAST;          -- 最后一个下标
    v_nums.EXISTS(i);     -- 检查下标 i 是否存在
    v_nums.DELETE;        -- 删除所有元素
    v_nums.DELETE(i);     -- 删除下标 i 的元素
    v_nums.DELETE(i, j);  -- 删除从 i 到 j 的元素
    v_nums.EXTEND;        -- 扩展一个 NULL 元素
    v_nums.EXTEND(n);     -- 扩展 n 个 NULL 元素
    v_nums.EXTEND(n, i);  -- 扩展 n 个值为 i 的元素
    v_nums.TRIM;          -- 删除最后一个元素
    v_nums.TRIM(n);       -- 删除最后 n 个元素
END;
/
```

## BULK COLLECT：批量查询

`BULK COLLECT` 是 Oracle 最强大的批量查询机制，将查询结果直接加载到集合中，避免逐行 FETCH 的性能开销。

```sql
DECLARE
    TYPE emp_name_list IS TABLE OF VARCHAR2(100);
    v_names emp_name_list;
BEGIN
    -- 批量查询所有员工姓名
    SELECT first_name || ' ' || last_name
    BULK COLLECT INTO v_names
    FROM employees
    WHERE department_id = 50;
    
    DBMS_OUTPUT.PUT_LINE('共 ' || v_names.COUNT || ' 名员工');
    
    FOR i IN 1..v_names.COUNT LOOP
        DBMS_OUTPUT.PUT_LINE(v_names(i));
    END LOOP;
END;
/

-- 批量查询多列
DECLARE
    TYPE emp_record IS RECORD (
        emp_id NUMBER,
        emp_name VARCHAR2(100),
        salary NUMBER
    );
    TYPE emp_table IS TABLE OF emp_record;
    v_emps emp_table;
BEGIN
    SELECT employee_id, first_name || ' ' || last_name, salary
    BULK COLLECT INTO v_emps
    FROM employees
    WHERE department_id = 50;
    
    FOR i IN 1..v_emps.COUNT LOOP
        DBMS_OUTPUT.PUT_LINE(v_emps(i).emp_name || ': ' || v_emps(i).salary);
    END LOOP;
END;
/
```

## FORALL：批量 DML

`FORALL` 是批量 DML 的关键，与 `BULK COLLECT` 配合使用可以实现高性能的批量操作：

```sql
-- 使用 FORALL 批量更新
DECLARE
    TYPE emp_id_list IS TABLE OF NUMBER INDEX BY PLS_INTEGER;
    TYPE salary_list IS TABLE OF NUMBER INDEX BY PLS_INTEGER;
    
    v_ids emp_id_list;
    v_salaries salary_list;
BEGIN
    -- 准备数据
    SELECT employee_id, salary
    BULK COLLECT INTO v_ids, v_salaries
    FROM employees
    WHERE department_id = 50;
    
    -- 批量更新：所有员工工资涨 10%
    FORALL i IN 1..v_ids.COUNT
        UPDATE employees
        SET salary = v_salaries(i) * 1.1
        WHERE employee_id = v_ids(i);
    
    DBMS_OUTPUT.PUT_LINE('更新了 ' || SQL%ROWCOUNT || ' 行');
    COMMIT;
END;
/

-- FORALL 的批量绑定
FORALL i IN 1..v_ids.COUNT
    INSERT INTO emp_history VALUES v_emp_list(i);
```

### FORALL 的执行模式

```sql
-- SAVE EXCEPTIONS：保存异常继续执行
DECLARE
    TYPE num_list IS TABLE OF NUMBER;
    v_nums num_list := num_list(1, 2, 3, 4, 5);
    v_errors NUMBER;
    TYPE err_type IS RECORD (error_index PLS_INTEGER, error_code PLS_INTEGER);
    TYPE err_list IS TABLE OF err_type;
    v_errs err_list;
BEGIN
    FORALL i IN 1..v_nums.COUNT SAVE EXCEPTIONS
        INSERT INTO test_table (id) VALUES (v_nums(i));
EXCEPTION
    WHEN OTHERS THEN
        v_errors := SQL%BULK_EXCEPTIONS.COUNT;
        DBMS_OUTPUT.PUT_LINE('共 ' || v_errors || ' 个错误');
        FOR i IN 1..v_errors LOOP
            DBMS_OUTPUT.PUT_LINE(
                '错误索引: ' || SQL%BULK_EXCEPTIONS(i).ERROR_INDEX ||
                ', 错误码: ' || SQL%BULK_EXCEPTIONS(i).ERROR_CODE
            );
        END LOOP;
END;
/
```

## 常见面试问题

**问：关联数组、嵌套表、VARRAY 有什么区别？**

关联数组下标灵活（整数或字符串），不支持持久化到数据库，是 PGA 内存结构，适合键值查找。嵌套表下标为整数，支持持久化为表列，可以存储在数据库中，但作为 PL/SQL 变量时是稀疏的。VARRAY 有固定上限（由 LIMIT 定义），下标为整数，适合"数量有上限"的场景，可持久化但大小受限。

**问：BULK COLLECT 和 FORALL 的作用是什么？**

BULK COLLECT 将 SQL 查询结果批量加载到 PL/SQL 集合中，避免逐行 FETCH 的性能开销（减少 context switch）。FORALL 将 PL/SQL 集合中的数据批量绑定到 DML 语句中执行，同样减少 context switch。两者配合使用可实现高性能的批量数据处理。

**问：VARRAY 有上限意味着什么？**

VARRAY 使用 `LIMIT` 定义最大元素数，超出上限会触发 ORA-06532 错误。适合已知最大元素数量的场景（如一周七天、一年十二个月）。嵌套表和关联数组没有硬性上限，嵌套表存储到数据库时最多 2GB。

## 总结

PL/SQL 集合类型提供了强大的内存数据处理能力。关联数组是最灵活的类型，适合键值查找；嵌套表支持持久化到数据库，适合需要长期存储集合数据的场景；VARRAY 有边界限制，适合元素数量固定的场景。BULK COLLECT 和 FORALL 是处理批量数据的关键机制，可以大幅提升 PL/SQL 程序的性能。
