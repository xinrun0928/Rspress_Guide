# HBase Java API

原生 HBase 提供了一套完整的 Java API，用于与 HBase 进行交互。理解这套 API，是使用 HBase 的基础。

虽然 Phoenix 等组件提供了 SQL 接口，但在很多场景下，原生 API 仍然是必要的选择——它更灵活，性能也更高。

## 环境准备

### Maven 依赖

```xml
<dependency>
    <groupId>org.apache.hbase</groupId>
    <artifactId>hbase-client</artifactId>
    <version>2.4.11</version>
</dependency>
```

### 连接配置

```java
// 创建配置
Configuration config = HBaseConfiguration.create();

// 设置 ZooKeeper 地址
config.set("hbase.zookeeper.quorum", "localhost");
config.set("hbase.zookeeper.property.clientPort", "2181");

// 设置其他参数
config.setInt("hbase.rpc.timeout", 60000);
config.setInt("hbase.client.operation.timeout", 60000);
config.setInt("hbase.client.scanner.timeout.period", 60000);

// 创建连接（推荐使用 ConnectionFactory）
Connection connection = ConnectionFactory.createConnection(config);

// 使用完毕后关闭连接
connection.close();
```

## 表操作

### 创建表

```java
public void createTable(Admin admin) throws IOException {
    // 定义表名
    TableName tableName = TableName.valueOf("users");
    
    // 定义列族
    ColumnFamilyDescriptor cf = ColumnFamilyDescriptorBuilder
        .newBuilder(Bytes.toBytes("info"))
        .setBlocksize(65536)           // Block 大小
        .setBloomFilterType(BloomType.ROWCOL)  // 布隆过滤器
        .setCompressionType(Compression.Algorithm.SNAPPY)  // 压缩
        .setTimeToLive(7776000)        // TTL：90天
        .build();
    
    // 创建表描述符
    TableDescriptor tableDescriptor = TableDescriptorBuilder
        .newBuilder(tableName)
        .setColumnFamily(cf)
        .build();
    
    // 创建表
    admin.createTable(tableDescriptor);
}
```

### 删除表

```java
public void deleteTable(Admin admin) throws IOException {
    TableName tableName = TableName.valueOf("users");
    
    // 删除前必须先禁用表
    admin.disableTable(tableName);
    admin.deleteTable(tableName);
}
```

### 修改表

```java
public void modifyTable(Admin admin) throws IOException {
    TableName tableName = TableName.valueOf("users");
    
    // 获取现有表描述符
    TableDescriptor existing = admin.getTableDescriptor(tableName);
    
    // 创建新的表描述符
    TableDescriptor modified = TableDescriptorBuilder.newBuilder(existing)
        .setColumnFamily(ColumnFamilyDescriptorBuilder
            .newBuilder(Bytes.toBytes("info"))
            .setBloomFilterType(BloomType.ROW)  // 修改布隆过滤器类型
            .build())
        .build();
    
    // 修改表
    admin.modifyTable(modified);
}
```

### 添加/删除列族

```java
public void modifyColumnFamily(Admin admin) throws IOException {
    TableName tableName = TableName.valueOf("users");
    
    // 添加新列族
    ColumnFamilyDescriptor newCf = ColumnFamilyDescriptorBuilder
        .newBuilder(Bytes.toBytes("order"))
        .setBlocksize(65536)
        .build();
    admin.addColumnFamily(tableName, newCf);
    
    // 删除列族
    admin.deleteColumnFamily(tableName, Bytes.toBytes("order"));
}
```

## 数据写入

### Put 操作

```java
public void putData(Table table) throws IOException {
    // 创建 Put 对象
    Put put = new Put(Bytes.toBytes("user_001"));
    
    // 添加列值
    put.addColumn(
        Bytes.toBytes("info"),           // Column Family
        Bytes.toBytes("name"),           // Column Qualifier
        Bytes.toBytes("张三")            // Value
    );
    
    put.addColumn(
        Bytes.toBytes("info"),
        Bytes.toBytes("age"),
        Bytes.toBytes(28)
    );
    
    // 设置时间戳（可选，默认使用服务器时间）
    put.setTimestamp(1704067200000L);
    
    // 写入
    table.put(put);
}
```

### 批量写入

```java
public void batchPut(Table table) throws IOException {
    List<Put> puts = new ArrayList<>();
    
    for (int i = 1; i <= 1000; i++) {
        Put put = new Put(Bytes.toBytes("user_" + String.format("%06d", i)));
        put.addColumn(
            Bytes.toBytes("info"),
            Bytes.toBytes("name"),
            Bytes.toBytes("用户" + i)
        );
        put.addColumn(
            Bytes.toBytes("info"),
            Bytes.toBytes("age"),
            Bytes.toBytes(20 + (i % 50))
        );
        puts.add(put);
    }
    
    // 批量写入
    table.put(puts);
}
```

### 异步写入

```java
public void asyncPut(Table table) throws IOException {
    // 创建异步表
    AsyncConnection asyncConnection = ConnectionFactory.createAsyncConnection(config).get();
    AsyncTable<AdvancedScanResultConsumer> asyncTable = 
        asyncConnection.getTableBuilder(TableName.valueOf("users"))
            .setWriteBufferSize(1024 * 1024)  // 1MB 缓冲区
            .build();
    
    // 异步写入
    CompletableFuture<Void> future = asyncTable.put(
        new Put(Bytes.toBytes("user_001"))
            .addColumn(Bytes.toBytes("info"), Bytes.toBytes("name"), Bytes.toBytes("张三"))
    );
    
    // 等待完成
    future.join();
}
```

## 数据读取

### Get 操作

```java
public void getData(Table table) throws IOException {
    // 创建 Get 对象
    Get get = new Get(Bytes.toBytes("user_001"));
    
    // 获取指定列族的所有列
    get.addFamily(Bytes.toBytes("info"));
    
    // 或获取指定列
    get.addColumn(Bytes.toBytes("info"), Bytes.toBytes("name"));
    
    // 获取所有版本
    get.readAllVersions();
    
    // 执行查询
    Result result = table.get(get);
    
    // 遍历结果
    for (Cell cell : result.listCells()) {
        String cf = Bytes.toString(CellUtil.cloneFamily(cell));
        String qualifier = Bytes.toString(CellUtil.cloneQualifier(cell));
        String value = Bytes.toString(CellUtil.cloneValue(cell));
        long timestamp = cell.getTimestamp();
        
        System.out.println(cf + ":" + qualifier + " = " + value + " (" + timestamp + ")");
    }
}
```

### Scan 操作

```java
public void scanData(Table table) throws IOException {
    // 创建 Scan 对象
    Scan scan = new Scan();
    
    // 设置起始和结束 RowKey
    scan.withStartRow(Bytes.toBytes("user_000001"));
    scan.withStopRow(Bytes.toBytes("user_000100"));
    
    // 设置列族和列
    scan.addFamily(Bytes.toBytes("info"));
    scan.addColumn(Bytes.toBytes("info"), Bytes.toBytes("name"));
    
    // 设置缓存行数（减少网络往返）
    scan.setCaching(100);
    
    // 设置每行返回的最大列数
    scan.setBatch(10);
    
    // 读取指定时间范围
    scan.setTimeRange(1704067200000L, 1704153600000L);
    
    // 执行扫描
    try (ResultScanner scanner = table.getScanner(scan)) {
        for (Result result : scanner) {
            // 处理每一行
            String rowKey = Bytes.toString(result.getRow());
            String name = Bytes.toString(result.getValue(
                Bytes.toBytes("info"), 
                Bytes.toBytes("name")
            ));
            System.out.println("RowKey: " + rowKey + ", Name: " + name);
        }
    }
}
```

## 过滤器

### 常用过滤器

```java
public void filterExamples(Table table) throws IOException {
    Scan scan = new Scan();
    
    // 1. 列前缀过滤器
    scan.setFilter(new ColumnPrefixFilter(Bytes.toBytes("na")));
    
    // 2. 行键前缀过滤器
    scan.setFilter(new PrefixFilter(Bytes.toBytes("user_")));
    
    // 3. 行范围过滤器
    scan.setFilter(new RowRangeFilter(
        CompareOperator.GREATER_OR_EQUAL,
        Bytes.toBytes("user_001"),
        true,  // 包含起始
        Bytes.toBytes("user_100"),
        false  // 不包含结束
    ));
    
    // 4. 值过滤器
    scan.setFilter(new ValueFilter(
        CompareOperator.EQUAL,
        new SubstringComparator("张三")
    ));
    
    // 5. 单列值过滤器
    scan.setFilter(new SingleColumnValueFilter(
        Bytes.toBytes("info"),
        Bytes.toBytes("age"),
        CompareOperator.GREATER,
        Bytes.toBytes(25)
    ));
    
    // 6. 多个过滤器组合
    FilterList filterList = new FilterList(FilterList.Operator.MUST_PASS_ALL);
    filterList.addFilter(new PrefixFilter(Bytes.toBytes("user_")));
    filterList.addFilter(new SingleColumnValueFilter(
        Bytes.toBytes("info"),
        Bytes.toBytes("age"),
        CompareOperator.GREATER,
        Bytes.toBytes(20)
    ));
    scan.setFilter(filterList);
    
    try (ResultScanner scanner = table.getScanner(scan)) {
        for (Result result : scanner) {
            // 处理结果
        }
    }
}
```

## 最佳实践

### 1. 连接复用

```java
// 不要每次操作都创建连接
public class HBaseConnectionManager {
    private static Connection connection;
    
    public static synchronized Connection getConnection() throws IOException {
        if (connection == null || connection.isClosed()) {
            Configuration config = HBaseConfiguration.create();
            config.set("hbase.zookeeper.quorum", "localhost");
            connection = ConnectionFactory.createConnection(config);
        }
        return connection;
    }
}
```

### 2. 使用连接池

```java
// 使用 BufferedMutator 进行异步批量写入
public void bufferedMutatorExample(TableName tableName, Connection connection) 
    throws IOException {
    
    // 创建 BufferedMutator
    BufferedMutatorParams params = new BufferedMutatorParams(tableName)
        .writeBufferSize(1024 * 1024)  // 1MB
        .listener((e, mutator) -> {
            System.err.println("Error on mutation: " + e.getMessage());
        });
    
    BufferedMutator mutator = connection.getBufferedMutator(params);
    
    try {
        // 写入数据
        for (int i = 0; i < 10000; i++) {
            Put put = new Put(Bytes.toBytes("user_" + i));
            put.addColumn(
                Bytes.toBytes("info"),
                Bytes.toBytes("name"),
                Bytes.toBytes("用户" + i)
            );
            mutator.mutate(put);
        }
    } finally {
        mutator.close();
    }
}
```

### 3. 合理设置缓存

```java
// 对于大量读取操作，增加缓存
Scan scan = new Scan();
scan.setCaching(1000);  // 每次从 RegionServer 获取 1000 行
scan.setCacheBlocks(true);  // 缓存到 BlockCache

// 对于只读一次的扫描，禁用缓存
Scan scan = new Scan();
scan.setCacheBlocks(false);
```

## 异常处理

```java
public void errorHandlingExample() {
    try (Table table = connection.getTable(TableName.valueOf("users"))) {
        Get get = new Get(Bytes.toBytes("user_001"));
        Result result = table.get(get);
        
        if (result.isEmpty()) {
            System.out.println("Row not found");
        }
    } catch (RetriesExhaustedWithDetailsException e) {
        // 处理重试耗尽的异常
        for (Throwable cause : e.getCauses()) {
            System.err.println("Cause: " + cause.getMessage());
        }
    } catch (IOException e) {
        // 处理 IO 异常
        e.printStackTrace();
    }
}
```
