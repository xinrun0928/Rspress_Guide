# PostgreSQL 存储过程 PL/pgSQL

PL/pgSQL 是 PostgreSQL 的过程式编程语言，允许你在数据库中编写复杂的业务逻辑。

---

## PL/pgSQL 基础

### 函数 vs 存储过程

```sql
-- 函数：返回数据
CREATE OR REPLACE FUNCTION get_user_count()
RETURNS INT AS $$
DECLARE
    count INT;
BEGIN
    SELECT COUNT(*) INTO count FROM users;
    RETURN count;
END;
$$ LANGUAGE plpgsql;

-- 存储过程（PostgreSQL 11+）
CREATE OR REPLACE PROCEDURE create_order(
    p_customer_id BIGINT,
    p_total NUMERIC
) AS $$
DECLARE
    v_order_id BIGINT;
BEGIN
    INSERT INTO orders (customer_id, total, created_at)
    VALUES (p_customer_id, p_total, NOW())
    RETURNING id INTO v_order_id;

    RAISE NOTICE 'Order created with ID: %', v_order_id;
END;
$$ LANGUAGE plpgsql;

-- 调用存储过程
CALL create_order(1, 99.99);
```

## 变量和类型

### 声明变量

```sql
CREATE OR REPLACE FUNCTION calculate_order(
    p_order_id BIGINT
)
RETURNS NUMERIC AS $$
DECLARE
    v_total NUMERIC(10, 2);      -- 声明变量
    v_tax NUMERIC(10, 2);
    v_discount NUMERIC(10, 2);
    v_final_total NUMERIC(10, 2);
BEGIN
    -- 赋值
    SELECT total INTO v_total FROM orders WHERE id = p_order_id;

    -- 计算
    v_tax := v_total * 0.1;
    v_discount := CASE
        WHEN v_total > 1000 THEN v_total * 0.1
        ELSE 0
    END;
    v_final_total := v_total + v_tax - v_discount;

    RETURN v_final_total;
END;
$$ LANGUAGE plpgsql;
```

### 记录类型

```sql
CREATE OR REPLACE FUNCTION get_order_details(
    p_order_id BIGINT
)
RETURNS TABLE (
    order_id BIGINT,
    customer_name VARCHAR,
    total NUMERIC
) AS $$
DECLARE
    rec RECORD;
BEGIN
    FOR rec IN
        SELECT o.id, c.name, o.total
        FROM orders o
        JOIN customers c ON o.customer_id = c.id
        WHERE o.id = p_order_id
    LOOP
        order_id := rec.id;
        customer_name := rec.name;
        total := rec.total;
        RETURN NEXT;
    END LOOP;
END;
$$ LANGUAGE plpgsql;
```

## 控制结构

### IF 语句

```sql
CREATE OR REPLACE FUNCTION get_discount(
    p_total NUMERIC
)
RETURNS NUMERIC AS $$
DECLARE
    discount NUMERIC;
BEGIN
    IF p_total > 1000 THEN
        discount := 0.15;
    ELSIF p_total > 500 THEN
        discount := 0.1;
    ELSIF p_total > 100 THEN
        discount := 0.05;
    ELSE
        discount := 0;
    END IF;

    RETURN discount;
END;
$$ LANGUAGE plpgsql;
```

### CASE 语句

```sql
CREATE OR REPLACE FUNCTION get_status_label(
    p_status VARCHAR
)
RETURNS VARCHAR AS $$
BEGIN
    RETURN CASE p_status
        WHEN 'pending' THEN '待处理'
        WHEN 'processing' THEN '处理中'
        WHEN 'shipped' THEN '已发货'
        WHEN 'delivered' THEN '已送达'
        WHEN 'cancelled' THEN '已取消'
        ELSE '未知状态'
    END;
END;
$$ LANGUAGE plpgsql;
```

### LOOP 循环

```sql
-- 简单循环
CREATE OR REPLACE FUNCTION generate_series_int(
    p_start INT,
    p_end INT
)
RETURNS TABLE (n INT) AS $$
DECLARE
    i INT := p_start;
BEGIN
    LOOP
        IF i > p_end THEN
            EXIT;
        END IF;
        n := i;
        RETURN NEXT;
        i := i + 1;
    END LOOP;
END;
$$ LANGUAGE plpgsql;

-- WHILE 循环
CREATE OR REPLACE FUNCTION fibonacci(
    p_count INT
)
RETURNS TABLE (n BIGINT) AS $$
DECLARE
    a BIGINT := 0;
    b BIGINT := 1;
    i INT := 0;
BEGIN
    WHILE i < p_count LOOP
        n := a;
        RETURN NEXT;
        a := b;
        b := a + b;
        i := i + 1;
    END LOOP;
END;
$$ LANGUAGE plpgsql;

-- FOR 循环（查询）
CREATE OR REPLACE FUNCTION get_top_customers(
    p_limit INT
)
RETURNS TABLE (
    customer_id BIGINT,
    total_amount NUMERIC
) AS $$
BEGIN
    FOR customer_id, total_amount IN
        SELECT customer_id, SUM(total)
        FROM orders
        GROUP BY customer_id
        ORDER BY SUM(total) DESC
        LIMIT p_limit
    LOOP
        RETURN NEXT;
    END LOOP;
END;
$$ LANGUAGE plpgsql;
```

## 异常处理

### 异常捕获

```sql
CREATE OR REPLACE FUNCTION safe_update_order(
    p_order_id BIGINT,
    p_status VARCHAR
)
RETURNS BOOLEAN AS $$
BEGIN
    UPDATE orders SET status = p_status WHERE id = p_order_id;

    IF NOT FOUND THEN
        RAISE NOTICE 'Order % not found', p_order_id;
        RETURN FALSE;
    END IF;

    RETURN TRUE;
EXCEPTION
    WHEN OTHERS THEN
        RAISE NOTICE 'Error updating order: %', SQLERRM;
        RETURN FALSE;
END;
$$ LANGUAGE plpgsql;
```

### 特定异常处理

```sql
CREATE OR REPLACE FUNCTION insert_user(
    p_name VARCHAR,
    p_email VARCHAR
)
RETURNS BIGINT AS $$
DECLARE
    v_user_id BIGINT;
BEGIN
    INSERT INTO users (name, email, created_at)
    VALUES (p_name, p_email, NOW())
    RETURNING id INTO v_user_id;

    RETURN v_user_id;
EXCEPTION
    WHEN unique_violation THEN
        RAISE NOTICE 'Email % already exists', p_email;
        RETURN -1;
    WHEN check_violation THEN
        RAISE NOTICE 'Check constraint violated';
        RETURN -2;
    WHEN foreign_key_violation THEN
        RAISE NOTICE 'Referenced record does not exist';
        RETURN -3;
END;
$$ LANGUAGE plpgsql;
```

## 事务控制

### 在存储过程中控制事务

```sql
-- PostgreSQL 11+ 支持存储过程中的事务控制
CREATE OR REPLACE PROCEDURE transfer_funds(
    p_from_account BIGINT,
    p_to_account BIGINT,
    p_amount NUMERIC
) AS $$
BEGIN
    -- 扣除金额
    UPDATE accounts SET balance = balance - p_amount
    WHERE id = p_from_account;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'Source account % not found', p_from_account;
    END IF;

    -- 检查余额
    IF (SELECT balance FROM accounts WHERE id = p_from_account) < 0 THEN
        ROLLBACK;
        RAISE EXCEPTION 'Insufficient funds';
    END IF;

    -- 增加金额
    UPDATE accounts SET balance = balance + p_amount
    WHERE id = p_to_account;

    IF NOT FOUND THEN
        ROLLBACK;
        RAISE EXCEPTION 'Destination account % not found', p_to_account;
    END IF;

    COMMIT;
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE;
END;
$$ LANGUAGE plpgsql;

-- 调用
CALL transfer_funds(1, 2, 100);
```

## 游标

### 显式游标

```sql
CREATE OR REPLACE FUNCTION process_orders(
    p_status VARCHAR
)
RETURNS VOID AS $$
DECLARE
    cur CURSOR FOR
        SELECT id, customer_id, total
        FROM orders
        WHERE status = p_status;
    rec RECORD;
BEGIN
    OPEN cur;

    LOOP
        FETCH cur INTO rec;
        EXIT WHEN NOT FOUND;

        -- 处理每条记录
        RAISE NOTICE 'Processing order: %, amount: %', rec.id, rec.total;

        -- 实际业务逻辑
        UPDATE orders SET status = 'processed' WHERE id = rec.id;
    END LOOP;

    CLOSE cur;
END;
$$ LANGUAGE plpgsql;
```

### 返回游标

```sql
CREATE OR REPLACE FUNCTION get_orders_cursor(
    p_status VARCHAR
)
RETURNS refcursor AS $$
DECLARE
    cur refcursor;
BEGIN
    OPEN cur FOR
        SELECT id, customer_id, total
        FROM orders
        WHERE status = p_status;

    RETURN cur;
END;
$$ LANGUAGE plpgsql;
```

## 总结

PL/pgSQL 是 PostgreSQL 的强大编程语言：

1. **函数和存储过程**：返回值或执行操作
2. **变量和类型**：声明、赋值、计算
3. **控制结构**：IF、CASE、LOOP、FOR
4. **异常处理**：捕获和处理错误
5. **事务控制**：在存储过程中管理事务
6. **游标**：逐行处理查询结果

PL/pgSQL 让复杂的业务逻辑可以在数据库层面处理，减少应用层的复杂度。
