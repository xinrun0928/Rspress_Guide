# PostgreSQL JSON/JSONB：灵活数据存储与查询

有时候，数据结构是**动态的**。

用户要添加自定义字段、产品规格随时变化、日志格式年年不同……传统关系型数据库的固定 Schema 在这些场景下显得笨拙——每次变化都要 ALTER TABLE。

PostgreSQL 的 **JSON/JSONB** 数据类型给出了一个优雅的答案：**在关系型数据库中存储半结构化数据**，既有 SQL 的能力，又有文档数据库的灵活。

## JSON vs JSONB：选哪个

### 核心区别

| 特性 | JSON | JSONB |
|---|---|---|
| 存储格式 | 存储原始 JSON 文本 | 存储解析后的二进制格式 |
| 读取速度 | 每次读取时解析 | 直接读取，更快 |
| 写入速度 | 快速（只存储文本）| 略慢（需要解析）|
| 索引支持 | 不支持 | 支持 GIN 索引 |
| 保留空格 | 保留原始格式 | 去除多余空格 |
| 保留顺序 | 保留 Key 顺序 | 保留插入顺序 |

**选择建议**：

- 需要**频繁查询** JSON 字段：`JSONB`
- 需要**保留原始格式**（如日志）：`JSON`
- 需要对 JSON **建索引**：`JSONB`

### 使用示例

```sql
-- JSON 类型：存储原始文本
CREATE TABLE api_logs (
    id SERIAL PRIMARY KEY,
    request_time TIMESTAMP,
    payload JSON
);

INSERT INTO api_logs VALUES (NOW(), 
    '{"method": "GET", "path": "/api/users", "status": 200}');

-- JSONB 类型：存储二进制格式
CREATE TABLE user_profiles (
    id SERIAL PRIMARY KEY,
    user_id INT,
    extra_data JSONB  -- 用户自定义字段
);

INSERT INTO user_profiles VALUES (1,
    '{"hobby": "摄影", "level": "pro", "tags": ["户外", "数码"]}');
```

## JSON 基础操作

### 创建 JSON

```sql
-- 使用字符串创建
SELECT '{"name": "张三", "age": 30}'::JSON;
SELECT '{"items": [1, 2, 3]}'::JSONB;

-- 使用 jsonb_build_object 函数
SELECT jsonb_build_object(
    'name', '张三',
    'age', 30,
    'skills', ARRAY['Java', 'Python']
);

-- 使用 jsonb_build_array 函数
SELECT jsonb_build_array(1, 2, 3, 4, 5);

-- 从行数据构建 JSON
SELECT jsonb_build_object(
    'id', id,
    'name', name,
    'email', email
) FROM users WHERE id = 1;
```

### 查询 JSON

```sql
-- 提取顶层 Key
SELECT '{"name": "张三", "age": 30}'::JSONB->>'name';  -- 返回 "张三"
SELECT '{"name": "张三", "age": 30}'::JSONB->'name';   -- 返回 "张三" (JSON)

-- 提取嵌套字段
SELECT '{"user": {"name": "张三", "address": {"city": "北京"}}}'::JSONB
    ->'user'->'address'->>'city';  -- 返回 "北京"

-- 使用 #> 路径操作符
SELECT '{"user": {"address": {"city": "北京"}}}'::JSONB #>> '{user,address,city}';

-- 提取数组元素
SELECT '["苹果", "香蕉", "橙子"]'::JSONB->1;  -- 返回 "香蕉"
SELECT '["苹果", "香蕉", "橙子"]'::JSONB->>1; -- 返回 "香蕉" (TEXT)
```

### JSON 函数

```sql
-- jsonb_object_keys: 获取所有 Key
SELECT jsonb_object_keys('{"a": 1, "b": 2, "c": 3}'::JSONB);
-- 输出：a, b, c

-- jsonb_array_length: 获取数组长度
SELECT jsonb_array_length('["苹果", "香蕉", "橙子"]'::JSONB);  -- 返回 3

-- jsonb_each: 展开 JSON 对象
SELECT * FROM jsonb_each('{"name": "张三", "age": 30}'::JSONB);
-- 输出：
--  key  | value
-- ------+-------
--  name | "张三"
--  age  | 30

-- jsonb_array_elements: 展开数组
SELECT * FROM jsonb_array_elements('["苹果", "香蕉", "橙子"]'::JSONB);
-- 输出：
--  value
-- -------
--  "苹果"
--  "香蕉"
--  "橙子"
```

### 修改 JSON

```sql
-- 使用 || 操作符合并
SELECT '{"name": "张三", "age": 30}'::JSONB 
    || '{"city": "北京"}'::JSONB;
-- 输出：{"name": "张三", "age": 30, "city": "北京"}

-- 删除 Key
SELECT '{"name": "张三", "age": 30, "city": "北京"}'::JSONB - 'age';
-- 输出：{"name": "张三", "city": "北京"}

-- 删除多个 Key
SELECT '{"name": "张三", "age": 30, "city": "北京"}'::JSONB - ARRAY['age', 'city'];

-- 更新 Key 值
SELECT '{"name": "张三", "age": 30}'::JSONB 
    || '{"age": 31}'::JSONB;

-- jsonb_set: 设置嵌套值
SELECT jsonb_set(
    '{"user": {"name": "张三"}}'::JSONB,
    '{user, name}',
    '"李四"'::JSONB
);
-- 输出：{"user": {"name": "李四"}}

-- jsonb_insert: 插入数组元素
SELECT jsonb_insert(
    '["苹果", "香蕉"]'::JSONB,
    '{1}',
    '"橙子"'::JSONB
);
-- 输出：["苹果", "橙子", "香蕉"]
```

## JSONB 高级查询

### jsonb_path_query：路径查询

```sql
-- 基本路径查询
SELECT jsonb_path_query(
    '{"user": {"orders": [{"id": 1, "amount": 100}, {"id": 2, "amount": 200}]}}'::JSONB,
    '$.user.orders[*].amount'
);
-- 输出：100, 200

-- 路径匹配过滤
SELECT jsonb_path_query(
    '{"orders": [{"status": "completed", "amount": 100}, {"status": "pending", "amount": 200}]}'::JSONB,
    '$.orders[*] ? (@.status == "completed")'
);
-- 输出：{"status": "completed", "amount": 100}

-- 路径中的变量
SELECT jsonb_path_query(
    '[{"name": "张三", "score": 90}, {"name": "李四", "score": 85}]'::JSONB,
    '$[*] ? (@.score > $minScore)'::JSONB,
    '{"minScore": 85}'
);
```

### jsonb_pretty：格式化输出

```sql
-- 格式化 JSON 输出（用于调试）
SELECT jsonb_pretty(
    '{"user": {"name": "张三", "orders": [{"id": 1, "items": ["A", "B"]}]}}'::JSONB
);

-- 输出：
-- [
--     {
--         "user": {
--             "name": "张三",
--             "orders": [
--                 {
--                     "id": 1,
--                     "items": ["A", "B"]
--                 }
--             ]
--         }
--     }
-- ]
```

### 复杂查询示例

```sql
-- 创建示例表
CREATE TABLE products (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    specs JSONB  -- 产品规格
);

INSERT INTO products VALUES
(1, '手机', '{"color": "黑色", "storage": "256GB", "ram": "8GB", "tags": ["5G", "旗舰"]}'),
(2, '平板', '{"color": "白色", "storage": "128GB", "ram": "6GB", "tags": ["性价比"]}'),
(3, '手表', '{"color": "银色", "material": "不锈钢", "tags": ["防水", "运动"]}');

-- 查询包含特定 tag 的产品
SELECT name, specs->>'color' as color
FROM products
WHERE specs @> '{"tags": ["5G"]}';

-- 查询某个规格的产品
SELECT name, specs->>'storage' as storage
FROM products
WHERE specs->>'ram' = '8GB';

-- 查询所有产品并展开 tags
SELECT name, tag
FROM products,
     jsonb_array_elements_text(specs->'tags') as tag
ORDER BY name, tag;
-- 输出：
--  name  |  tag
-- -------+-------
--  手表   | 运动
--  手表   | 防水
--  手机   | 5G
--  手机   | 旗舰
--  平板   | 性价比

-- 统计每个 tag 的产品数量
SELECT tag, COUNT(*) as count
FROM products,
     jsonb_array_elements_text(specs->'tags') as tag
GROUP BY tag
ORDER BY count DESC;
```

## JSONB 索引

### GIN 索引

JSONB 最常用的索引类型：

```sql
-- 创建 GIN 索引（包含所有 Key）
CREATE INDEX idx_products_specs ON products USING GIN (specs);

-- 使用 @> 操作符查询（会使用索引）
SELECT * FROM products WHERE specs @> '{"color": "黑色"}';
SELECT * FROM products WHERE specs @> '{"tags": ["5G"]}';

-- 使用 ? 操作符查询（Key 存在）
SELECT * FROM products WHERE specs ? 'ram';
SELECT * FROM products WHERE specs ?| ARRAY['ram', 'storage'];  -- 任一存在
SELECT * FROM products WHERE specs ?& ARRAY['ram', 'storage'];  -- 同时存在
```

### 表达式索引

```sql
-- 为特定字段创建表达式索引
CREATE INDEX idx_products_color ON products ((specs->>'color'));

-- 为嵌套路径创建索引
CREATE INDEX idx_products_user_orders 
    ON orders USING GIN ((data #'>{user,orders}'));
```

### 性能对比

```sql
-- 无索引查询
EXPLAIN SELECT * FROM products WHERE specs @> '{"color": "黑色"}';
-- Seq Scan on products  (cost=0.00..100.00 rows=10 width=200)

-- 有 GIN 索引
EXPLAIN SELECT * FROM products WHERE specs @> '{"color": "黑色"}';
-- Bitmap Index Scan on idx_products_specs
--   Index Cond: (specs @> '{"color": "黑色"}'::JSONB)
```

## 实战案例

### 案例一：用户画像系统

```sql
-- 用户画像表：支持动态扩展字段
CREATE TABLE user_profiles (
    id SERIAL PRIMARY KEY,
    user_id INT UNIQUE NOT NULL,
    base_info JSONB,      -- 基础信息
    behavior JSONB,        -- 行为数据
    preferences JSONB      -- 偏好设置
);

-- 插入数据
INSERT INTO user_profiles VALUES
(1, 1001,
 '{"name": "张三", "age": 30, "gender": "男"}',
 '{"login_days": 30, "total_orders": 50, "avg_amount": 200}',
 '{"category": ["数码", "图书"], "brand": ["Apple", "小米"]}');

-- 查询用户行为
SELECT 
    user_id,
    base_info->>'name' as name,
    behavior->>'login_days' as login_days,
    behavior->>'total_orders' as total_orders
FROM user_profiles
WHERE behavior->>'total_orders'::INT > 30;

-- 查找有特定偏好的用户
SELECT user_id, preferences->>'category' as category
FROM user_profiles
WHERE preferences @> '{"category": ["数码"]}';
```

### 案例二：日志分析系统

```sql
-- 访问日志表
CREATE TABLE access_logs (
    id BIGSERIAL PRIMARY KEY,
    log_time TIMESTAMP,
    log_data JSONB
);

-- 批量插入日志
INSERT INTO access_logs (log_time, log_data)
SELECT 
    NOW() - (random() * interval '7 days'),
    jsonb_build_object(
        'ip', '192.168.1.' || (random() * 255)::INT,
        'method', (ARRAY['GET', 'POST', 'PUT', 'DELETE'])[floor(random() * 4 + 1)::INT],
        'path', (ARRAY['/api/users', '/api/products', '/api/orders'])[floor(random() * 3 + 1)::INT],
        'status', (ARRAY[200, 200, 200, 404, 500])[floor(random() * 5 + 1)::INT],
        'response_time', (random() * 1000)::INT,
        'user_agent', (ARRAY['Chrome', 'Firefox', 'Safari'])[floor(random() * 3 + 1)::INT]
    )
FROM generate_series(1, 10000);

-- 分析访问趋势
SELECT 
    date_trunc('hour', log_time) as hour,
    log_data->>'path' as path,
    COUNT(*) as requests,
    AVG((log_data->>'response_time')::INT) as avg_response_time
FROM access_logs
WHERE log_data->>'status' = '200'
GROUP BY 1, 2
ORDER BY 1, 2;

-- 查找异常请求
SELECT * FROM access_logs
WHERE (log_data->>'status')::INT >= 500
   OR (log_data->>'response_time')::INT > 5000;
```

## 最佳实践

### 性能优化建议

```sql
-- 1. 避免在 JSON 中存储所有数据
-- 应该：关系字段保持为列，动态字段使用 JSON
CREATE TABLE orders (
    order_id SERIAL PRIMARY KEY,
    customer_id INT,
    order_date TIMESTAMP,
    total_amount DECIMAL(10,2),
    items JSONB  -- 订单明细（动态结构）
);

-- 2. 常用查询字段建索引
CREATE INDEX idx_orders_items_product 
    ON orders USING GIN ((items));

-- 3. 定期 VACUUM
VACUUM ANALYZE orders;

-- 4. 控制 JSON 字段大小
-- JSONB 会预解析和存储，过大的 JSON 会影响性能
-- 建议单条 JSON 不超过 1MB
```

### 数据校验

```sql
-- 使用 JSON Schema 校验（PG 14+）
CREATE TABLE api_data (
    id SERIAL PRIMARY KEY,
    payload JSONB
        CONSTRAINT payload_schema CHECK (
            payload IS NOT NULL AND 
            payload @> '{"name": "test"}'
        )
);

-- 使用函数验证 JSON 结构
CREATE OR REPLACE FUNCTION validate_user_data(data JSONB)
RETURNS BOOLEAN AS $$
BEGIN
    RETURN 
        data ? 'name' AND
        data ? 'email' AND
        data->>'email' ~* '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}$';
END;
$$ LANGUAGE plpgsql IMMUTABLE;

-- 应用校验
SELECT * FROM user_profiles 
WHERE validate_user_data(behavior);
```

## 总结

PostgreSQL 的 JSON/JSONB 为关系型数据库带来了文档数据库的灵活性：

- **JSONB**：二进制存储，支持索引，推荐用于查询场景
- **路径查询**：`jsonb_path_query` 支持强大的 JSONPath 语法
- **GIN 索引**：让 JSON 查询也能高效执行
- **jsonb_pretty**：格式化输出，方便调试

当你需要在关系型数据库中处理半结构化数据时，JSON/JSONB 是最好的选择——**既有 SQL 的能力，又有 NoSQL 的灵活**。
