# PostgreSQL 性能监控

性能监控是数据库运维的核心。没有监控，就没有优化的依据。

PostgreSQL 提供了丰富的监控视图和工具，帮助你了解数据库的运行状态。

---

## 核心监控视图

### pg_stat_activity

```sql
-- 查看当前活动连接
SELECT
    pid,
    usename,
    application_name,
    state,
    query,
    query_start,
    state_change,
    wait_event_type,
    wait_event,
    (NOW() - query_start)::TEXT AS duration,
    (NOW() - state_change)::TEXT AS idle_time
FROM pg_stat_activity
WHERE state != 'idle'
ORDER BY query_start;
```

### pg_stat_database

```sql
-- 查看数据库统计
SELECT
    datname,
    numbackends AS connections,
    xact_commit AS commits,
    xact_rollback AS rollbacks,
    blks_hit,
    blks_read,
    round(100.0 * blks_hit / NULLIF(blks_hit + blks_read, 0), 2) AS cache_hit_ratio,
    tup_returned,
    tup_fetched,
    tup_inserted,
    tup_updated,
    tup_deleted,
    conflicts,
    temp_files,
    temp_bytes,
    deadlocks
FROM pg_stat_database
WHERE datname = current_database();
```

### pg_stat_user_tables

```sql
-- 查看表的访问统计
SELECT
    schemaname,
    relname AS table_name,
    seq_scan,
    seq_tup_read,
    idx_scan,
    idx_tup_fetch,
    n_tup_ins,
    n_tup_upd,
    n_tup_del,
    n_live_tup,
    n_dead_tup,
    n_mod_since_analyze,
    last_analyze,
    last_autoanalyze,
    last_vacuum,
    last_autovacuum,
    pg_size_pretty(pg_relation_size(relid)) AS table_size
FROM pg_stat_user_tables
WHERE schemaname = 'public'
ORDER BY seq_scan DESC;
```

### pg_stat_user_indexes

```sql
-- 查看索引使用统计
SELECT
    schemaname,
    relname AS table_name,
    indexrelname AS index_name,
    idx_scan,
    idx_tup_read,
    idx_tup_fetch,
    idx_tup_read - idx_tup_fetch AS heap_reads,
    pg_size_pretty(pg_relation_size(indexrelid)) AS index_size
FROM pg_stat_user_indexes
WHERE schemaname = 'public'
ORDER BY idx_scan;
```

## 连接监控

### 连接状态

```sql
-- 查看连接分布
SELECT
    state,
    COUNT(*) AS count
FROM pg_stat_activity
GROUP BY state;

-- 查看等待锁的连接
SELECT
    a.pid,
    a.usename,
    a.query,
    a.wait_event_type,
    a.wait_event,
    l.locktype,
    l.mode
FROM pg_stat_activity a
JOIN pg_locks l ON a.pid = l.pid
WHERE NOT l.granted;
```

### 连接限制

```sql
-- 查看最大连接数和使用情况
SELECT
    current_setting('max_connections')::INT AS max_conn,
    (SELECT COUNT(*) FROM pg_stat_activity) AS current_conn,
    current_setting('max_connections')::INT - (SELECT COUNT(*) FROM pg_stat_activity) AS available_conn;
```

## 缓存监控

### 缓存命中率

```sql
-- 查看缓存命中率
SELECT
    datname,
    blks_hit,
    blks_read,
    CASE WHEN blks_read + blks_hit > 0
         THEN round(100.0 * blks_hit / (blks_read + blks_hit), 2)
         ELSE 0 END AS hit_ratio,
    pg_size_pretty(blks_hit * current_setting('block_size')::INT) AS cache_hits_size,
    pg_size_pretty(blks_read * current_setting('block_size')::INT) AS disk_reads_size
FROM pg_stat_database;
```

### 共享缓冲区状态

```sql
-- 查看共享缓冲区使用情况
SELECT
    (SELECT count(*) FROM pg_stat_activity) AS active_connections,
    (SELECT sum(blks_hit) FROM pg_stat_database) AS total_cache_hits,
    (SELECT sum(blks_read) FROM pg_stat_database) AS total_disk_reads,
    current_setting('shared_buffers') AS shared_buffers;
```

## VACUUM 监控

### 膨胀监控

```sql
-- 查看表膨胀情况
SELECT
    schemaname,
    relname AS table_name,
    n_live_tup,
    n_dead_tup,
    CASE WHEN n_live_tup + n_dead_tup > 0
         THEN round(100.0 * n_dead_tup / (n_live_tup + n_dead_tup), 2)
         ELSE 0 END AS dead_ratio,
    last_autovacuum,
    pg_size_pretty(pg_relation_size(relid)) AS table_size
FROM pg_stat_user_tables
WHERE n_dead_tup > 1000
ORDER BY n_dead_tup DESC;
```

### VACUUM 进度

```sql
-- 查看 VACUUM 进度（PostgreSQL 13+）
SELECT
    pid,
    query,
    phase,
    pg_size_pretty(heap_blks_total * current_setting('block_size')::INT) AS total_size,
    pg_size_pretty(heap_blks_scanned * current_setting('block_size')::INT) AS scanned_size,
    100.0 * heap_blks_scanned / NULLIF(heap_blks_total, 0) AS progress
FROM pg_stat_progress_vacuum;
```

## WAL 监控

### WAL 使用

```sql
-- 查看 WAL 状态
SELECT
    pg_current_wal_lsn() AS current_lsn,
    pg_wal_lsn_diff(pg_current_wal_lsn(), '0/0'::pg_lsn) AS wal_bytes;
```

### WAL 统计

```sql
-- 查看 WAL 写入统计
SELECT
    total_size,
    total_writes,
    total_buffers_used
FROM pg_stat_wal;
```

## 复制监控

### 流复制状态

```sql
-- 查看复制状态
SELECT
    client_addr,
    state,
    sent_lsn,
    write_lsn,
    flush_lsn,
    replay_lsn,
    (sent_lsn - replay_lsn) AS replication_lag
FROM pg_stat_replication;
```

### 复制延迟

```sql
-- 查看复制延迟（时间）
SELECT
    client_addr,
    state,
    (NOW() - reply_time)::TEXT AS reply_time,
    (sent_lsn - replay_lsn) AS lsn_lag
FROM pg_stat_replication
ORDER BY lsn_lag DESC;
```

## 性能监控工具

### pg_stat_statements

```sql
-- 安装扩展
CREATE EXTENSION pg_stat_statements;

-- 查看最慢的查询
SELECT
    query,
    calls,
    total_exec_time / 1000 AS total_time_sec,
    mean_exec_time AS mean_time_ms,
    rows,
    shared_blks_hit,
    shared_blks_read,
    100.0 * shared_blks_hit / NULLIF(shared_blks_hit + shared_blks_read, 0) AS cache_hit_ratio
FROM pg_stat_statements
ORDER BY total_exec_time DESC
LIMIT 10;
```

## 监控脚本示例

### 基础监控脚本

```sql
-- 创建一个监控视图
CREATE OR REPLACE VIEW db_health AS
SELECT
    'connections' AS metric,
    (SELECT COUNT(*) FROM pg_stat_activity) AS value
UNION ALL
SELECT
    'cache_hit_ratio' AS metric,
    (SELECT round(100.0 * sum(blks_hit) / sum(blks_hit + blks_read), 2)
     FROM pg_stat_database WHERE datname = current_database()) AS value
UNION ALL
SELECT
    'active_transactions' AS metric,
    (SELECT COUNT(*) FROM pg_stat_activity WHERE state = 'active') AS value
UNION ALL
SELECT
    'waiting_transactions' AS metric,
    (SELECT COUNT(*) FROM pg_stat_activity WHERE wait_event IS NOT NULL) AS value;

-- 查看健康状态
SELECT * FROM db_health;
```

### 告警设置

```sql
-- 连接数告警
SELECT
    CASE WHEN COUNT(*) > 100 THEN 'ALERT: High connections' ELSE 'OK' END AS status
FROM pg_stat_activity;

-- 复制延迟告警
SELECT
    CASE WHEN (sent_lsn - replay_lsn) > 1024 * 1024 * 100 THEN 'ALERT: Replication lag'
         ELSE 'OK' END AS status
FROM pg_stat_replication;
```

## 总结

PostgreSQL 提供了丰富的监控能力：

1. **活动监控**：pg_stat_activity 查看当前连接和查询
2. **数据库统计**：pg_stat_database 查看整体运行状态
3. **表和索引统计**：pg_stat_user_tables/indexes 了解访问模式
4. **VACUUM 监控**：pg_stat_progress_vacuum 跟踪垃圾回收进度
5. **复制监控**：pg_stat_replication 了解主从延迟

建立完善的监控体系，是保证 PostgreSQL 高性能运行的基础。
