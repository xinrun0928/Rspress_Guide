## Camunda EngineпјҲжөҒзЁӢеј•ж“Һпјү

### ж ёеҝғз»„д»¶

Camunda Engine жҳҜж•ҙдёӘе№іеҸ°зҡ„ж ёеҝғпјҢиҙҹиҙЈ BPMN жөҒзЁӢзҡ„жү§иЎҢгҖӮ

```java
// иҺ·еҸ–жөҒзЁӢеј•ж“Һ
ProcessEngine processEngine = ProcessEngineConfiguration
    .createStandaloneProcessEngineConfiguration()
    .buildProcessEngine();

// иҺ·еҸ–ж ёеҝғжңҚеҠЎ
RuntimeService runtimeService = processEngine.getRuntimeService();
TaskService taskService = processEngine.getTaskService();
RepositoryService repositoryService = processEngine.getRepositoryService();
HistoryService historyService = processEngine.getHistoryService();
ProcessManagementService processManagementService = processEngine.getProcessManagementService();
```

### еј•ж“Һй…ҚзҪ®

Camunda ж”ҜжҢҒеӨҡз§Қй…ҚзҪ®ж–№ејҸпјҡ

```java
// ж–№ејҸ1пјҡзј–зЁӢејҸй…ҚзҪ®
ProcessEngineConfiguration config = ProcessEngineConfiguration
    .createStandaloneProcessEngineConfiguration()
    .setJdbcUrl("jdbc:mysql://localhost:3306/camunda")
    .setJdbcDriver("com.mysql.cj.jdbc.Driver")
    .setJdbcUsername("root")
    .setJdbcPassword("password")
    .setDatabaseSchemaUpdate(ProcessEngineConfiguration.DB_SCHEMA_UPDATE_TRUE)
    .setJobExecutorActivate(true)
    .setAsyncExecutorEnabled(true);

// ж–№ејҸ2пјҡXML й…ҚзҪ®
ProcessEngine processEngine = ProcessEngineConfiguration
    .createProcessEngineConfigurationFromResource("camunda.cfg.xml")
    .buildProcessEngine();
```

```xml
<!-- camunda.cfg.xml -->
<beans xmlns="http://www.springframework.org/schema/beans">
    <bean id="processEngineConfiguration" 
          class="org.camunda.bpm.engine.impl.cfg.StandaloneProcessEngineConfiguration">
        <property name="jdbcUrl" value="jdbc:mysql://localhost:3306/camunda"/>
        <property name="jdbcDriver" value="com.mysql.cj.jdbc.Driver"/>
        <property name="jdbcUsername" value="root"/>
        <property name="jdbcPassword" value="password"/>
        <property name="databaseSchemaUpdate" value="true"/>
    </bean>
</beans>
```

### ж ёеҝғжңҚеҠЎдёҖи§Ҳ

|---|---|---|

```java
/**
 * еӨ–йғЁд»»еҠЎе®ўжҲ·з«Ҝ
 * еӨ–йғЁзі»з»ҹдҪҝз”ЁиҝҷдёӘе®ўжҲ·з«Ҝд»Һ Camunda иҺ·еҸ–д»»еҠЎе№¶еӨ„зҗҶ
 */
public class ExternalTaskWorker {
    
    private final ProcessEngine processEngine;
    private final String workerId;
    
    public ExternalTaskWorker(ProcessEngine processEngine, String workerId) {
        this.processEngine = processEngine;
        this.workerId = workerId;
    }
    
    /**
     * жӢүеҸ–е№¶й”Ғе®ҡд»»еҠЎ
     */
    public void fetchAndHandleTasks() {
        ExternalTaskService externalTaskService = processEngine.getExternalTaskService();
        
        // жӢүеҸ–д»»еҠЎ
        List<LockedExternalTask> tasks = externalTaskService.fetchAndLock(5, workerId)
            .topic("inventory-check", Duration.ofMinutes(5))
            .variables("productId", "quantity")
            .execute();
        
        for (LockedExternalTask task : tasks) {
            try {
                // еӨ„зҗҶдёҡеҠЎйҖ»иҫ‘
                handleInventoryCheck(task);
                
                // е®ҢжҲҗд»»еҠЎ
                externalTaskService.complete(task.getId(), workerId);
                
            } catch (Exception e) {
                // еӨ„зҗҶеӨұиҙҘпјҢжҠҘе‘Ҡй”ҷиҜҜ
                externalTaskService.handleFailure(
                    task.getId(),
                    workerId,
                    "еә“еӯҳжЈҖжҹҘеӨұиҙҘ: " + e.getMessage(),
                    3,  // йҮҚиҜ•ж¬Ўж•°
                    Duration.ofMinutes(1).toMillis()  // йҮҚиҜ•й—ҙйҡ”
                );
            }
        }
    }
    
    /**
     * дёҡеҠЎеӨ„зҗҶйҖ»иҫ‘
     */
    private void handleInventoryCheck(LockedExternalTask task) {
        String productId = (String) task.getVariables().get("productId");
        Integer quantity = (Integer) task.getVariables().get("quantity");
        
        // и°ғз”Ёеә“еӯҳзі»з»ҹжЈҖжҹҘ
        InventoryCheckResult result = inventoryService.check(productId, quantity);
        
        // и®ҫзҪ®иҫ“еҮәеҸҳйҮҸ
        Map<String, Object> variables = new HashMap<>();
        variables.put("inStock", result.isAvailable());
        variables.put("stockQuantity", result.getAvailableQuantity());
        
        // жіЁж„ҸпјҡйңҖиҰҒеңЁ complete ж—¶дј е…Ҙ
    }
}
```

### йӣҶзҫӨй…ҚзҪ®иҰҒзӮ№

```java
/**
 * йӣҶзҫӨзҺҜеўғдёӢзҡ„еј•ж“Һй…ҚзҪ®
 */
ProcessEngineConfiguration clusterConfig = ProcessEngineConfiguration
    .createProcessEngineConfigurationFromResource("clustered-camunda.cfg.xml")
    // еҗҜз”ЁдҪңдёҡжү§иЎҢеҷЁ
    .setJobExecutorActivate(true)
    .setAsyncExecutorEnabled(true)
    // дҪңдёҡиҺ·еҸ–й”Ғе®ҡж—¶й—ҙпјҲйҳІжӯўйҮҚеӨҚжү§иЎҢпјү
    .setJobExecutorLockTime(TimeDuration.ofMinutes(5))
    // ж•°жҚ®еә“schemaиҮӘеҠЁжӣҙж–°
    .setDatabaseSchemaUpdate(DB_SCHEMA_UPDATE_TRUE)
    // жөҒзЁӢеј•ж“ҺеҗҚз§°пјҲйӣҶзҫӨдёӯе”ҜдёҖпјү
    .setProcessEngineName("cluster-node-1");
```

## з•ҷз»ҷдҪ зҡ„й—®йўҳ

еҒҮи®ҫдҪ иҰҒжҠҠ Camunda йғЁзҪІеҲ°з”ҹдә§зҺҜеўғпјҢйңҖиҰҒж”ҜжҢҒд»ҘдёӢйңҖжұӮпјҡ

1. жҜҸеӨ©еӨ„зҗҶ 10 дёҮ+ жөҒзЁӢе®һдҫӢ
2. йңҖиҰҒ 99.9% зҡ„еҸҜз”ЁжҖ§
3. еӨҡдёӘдёҡеҠЎзі»з»ҹйңҖиҰҒи°ғз”Ёе·ҘдҪңжөҒ API

**й—®йўҳжқҘдәҶпјҡ**

1. Web еә”з”ЁпјҲCockpit/Tasklistпјүеә”иҜҘе’Ңеј•ж“ҺеҲҶејҖйғЁзҪІиҝҳжҳҜж”ҫеңЁдёҖиө·пјҹеҲҶејҖйғЁзҪІжңүд»Җд№ҲеҘҪеӨ„пјҹ
2. йӣҶзҫӨзҺҜеўғдёӢпјҢеҰӮжһңдёӨдёӘиҠӮзӮ№еҗҢж—¶жҹҘиҜўеҲ°дёҖдёӘеҫ…жү§иЎҢзҡ„дҪңдёҡпјҢдјҡеҸ‘з”ҹд»Җд№ҲпјҹCamunda жҳҜжҖҺд№ҲйҒҝе…ҚйҮҚеӨҚжү§иЎҢзҡ„пјҹ
3. REST API жҡҙйңІз»ҷеӨҡдёӘдёҡеҠЎзі»з»ҹи°ғз”ЁпјҢеҰӮдҪ•еҒҡжқғйҷҗжҺ§еҲ¶е’ҢжөҒйҮҸйҷҗеҲ¶пјҹ

иҝҷдёүдёӘй—®йўҳж¶үеҸҠеҲ°**жһ¶жһ„и®ҫи®Ў**гҖҒ**еҲҶеёғејҸдёҖиҮҙжҖ§**е’Ң**API е®үе…Ё**пјҢжҳҜз”ҹдә§зҺҜеўғйғЁзҪІзҡ„ж ёеҝғиҖғйҮҸгҖӮ
