## 一、Redis ZSet 排行榜

### ZSet 数据结构

| 命令 | 说明 | 时间复杂度 |
|------|------|------------|
| ZADD | 添加/更新分数 | O(log N) |
| ZINCRBY | 增加分数 | O(log N) |
| ZRANK | 获取排名 | O(log N) |
| ZREVRANK | 获取倒序排名 | O(log N) |
| ZSCORE | 获取分数 | O(1) |
| ZRANGE | 获取排名范围 | O(log N + M) |
| ZREVRANGE | 获取倒序范围 | O(log N + M) |

### 基本操作

```java
@Service
public class LeaderboardService {
    @Autowired
    private RedisTemplate<String, String> redisTemplate;

    private static final String LEADERBOARD_KEY = "leaderboard:game:";

    // 更新分数
    public void updateScore(Long userId, int delta) {
        String key = LEADERBOARD_KEY + "daily"; // 日榜
        redisTemplate.opsForZSet().incrementScore(key, userId.toString(), delta);
    }

    // 获取排名
    public Long getRank(Long userId) {
        String key = LEADERBOARD_KEY + "daily";
        Long rank = redisTemplate.opsForZSet().reverseRank(key, userId.toString());
        return rank != null ? rank + 1 : null; // 转换为从 1 开始
    }

    // 获取 Top N
    public Set<ZSetOperations.TypedTuple<String>> getTopN(int n) {
        String key = LEADERBOARD_KEY + "daily";
        return redisTemplate.opsForZSet().reverseRangeWithScores(key, 0, n - 1);
    }

    // 获取用户周边
    public Set<String> getAround(Long userId, int range) {
        String key = LEADERBOARD_KEY + "daily";
        Long rank = redisTemplate.opsForZSet().reverseRank(key, userId.toString());
        if (rank == null) return Collections.emptySet();

        long start = Math.max(0, rank - range);
        long end = rank + range;

        return redisTemplate.opsForZSet().reverseRange(key, start, end);
    }
}
```

## 二、多维度排行榜

```java
@Service
public class MultiLeaderboardService {
    @Autowired
    private RedisTemplate<String, String> redisTemplate;

    // 同时更新多个排行榜
    public void updateAll(Long userId, LeaderboardUpdate update) {
        // 1. 总战力榜
        updateLeaderboard("rank:total", userId, update.getTotalScore());

        // 2. 日榜
        updateLeaderboard("rank:daily", userId, update.getDailyScore());

        // 3. 周榜
        updateLeaderboard("rank:weekly", userId, update.getWeeklyScore());

        // 4. 月榜
        updateLeaderboard("rank:monthly", userId, update.getMonthlyScore());
    }

    private void updateLeaderboard(String key, Long userId, double score) {
        redisTemplate.opsForZSet().add(key, userId.toString(), score);
    }
}
```

## 三、定时快照

```java
@Service
public class SnapshotService {
    @Autowired
    private RedisTemplate<String, String> redisTemplate;

    // 每日凌晨快照
    @Scheduled(cron = "0 0 0 * * ?")
    public void dailySnapshot() {
        // 1. 获取日榜前 1000 名
        String dailyKey = "rank:daily";
        Set<ZSetOperations.TypedTuple<String>> topUsers =
            redisTemplate.opsForZSet().reverseRangeWithScores(dailyKey, 0, 999);

        // 2. 写入历史表
        LocalDate yesterday = LocalDate.now().minusDays(1);
        String snapshotKey = "rank:snapshot:" + yesterday;

        if (topUsers != null) {
            for (ZSetOperations.TypedTuple<String> tuple : topUsers) {
                String userId = tuple.getValue();
                Double score = tuple.getScore();
                if (userId != null && score != null) {
                    redisTemplate.opsForZSet().add(snapshotKey, userId, score);
                }
            }
        }

        // 3. 保留 30 天
        redisTemplate.expire(snapshotKey, Duration.ofDays(30));

        // 4. 清空日榜
        redisTemplate.delete(dailyKey);
    }
}
```

## 四、数据持久化

```sql
CREATE TABLE leaderboard_snapshot (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    rank_type VARCHAR(32) NOT NULL,    -- daily/weekly/monthly/total
    rank_date DATE NOT NULL,
    score DECIMAL(20,2) NOT NULL,
    `rank` INT NOT NULL,
    created_at DATETIME,
    INDEX idx_type_date (rank_type, rank_date),
    INDEX idx_user (user_id, rank_type, rank_date)
) ENGINE=InnoDB;
```
