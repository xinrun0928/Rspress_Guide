/**
 * 数据库 Session 存储
 */
public class DatabaseSessionManager implements SessionStorable, BranchSessionStorable {
    
    private DataSource dataSource;
    
    @Override
    public void storeGlobalSession(GlobalSession globalSession) {
        // 1. 序列化 GlobalSession
        String data = serialize(globalSession);
        
        // 2. 写入数据库
        String sql = "INSERT INTO global_table (xid, transaction_id, status, " +
            "application_id, transaction_service_group, transaction_name, " +
            "timeout, begin_time, application_data) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)";
        
        jdbcTemplate.update(sql, 
            globalSession.getXid(),
            globalSession.getTransactionId(),
            globalSession.getStatus().getCode(),
            globalSession.getApplicationId(),
            globalSession.getTransactionServiceGroup(),
            globalSession.getTransactionName(),
            globalSession.getTimeout(),
            globalSession.getBeginTime(),
            data
        );
    }
    
    @Override
    public GlobalSession findGlobalSession(String xid) {
        String sql = "SELECT * FROM global_table WHERE xid = ?";
        return jdbcTemplate.queryForObject(sql, (rs, rowNum) -> {
            return deserialize(rs.getString("application_data"));
        }, xid);
    }
}

### 锁申请与释放

```java
/**
 * 全局锁管理
 */
public class GlobalLockManager {
    
    /**
     * 尝试获取锁
     */
    public boolean tryLock(String xid, long branchId, 
                          String resourceId, String lockKey) {
        // 1. 检查锁是否已被其他事务持有
        LockDO existingLock = findLock(lockKey);
        
        if (existingLock == null) {
            // 锁未被持有，直接获取
            return acquireLock(xid, branchId, resourceId, lockKey);
        }
        
        if (existingLock.getXid().equals(xid)) {
            // 同一事务，直接成功
            return true;
        }
        
        // 锁被其他事务持有
        return false;
    }
    
    /**
     * 释放锁
     */
    public void unLock(String xid, long branchId, String lockKey) {
        // 删除锁记录
        String sql = "DELETE FROM lock_table WHERE xid = ? AND branch_id = ? AND row_key = ?";
        jdbcTemplate.update(sql, xid, branchId, lockKey);
    }
    
    /**
     * 检查是否可以安全回滚
     */
    public boolean isLockable(String xid, String lockKey) {
        // 检查是否有其他事务持有该锁
        LockDO lock = findLock(lockKey);
        
        // 如果没有其他事务持有，可以安全回滚
        return lock == null || lock.getXid().equals(xid);
    }
}
```

## 故障恢复

### 服务故障恢复

```java
/**
 * 故障恢复机制
 */

// Seata Server 重启后，需要恢复未决事务

/**
 * 恢复流程
 */
public class TransactionCoordinatorHealthChecker {
    
    /**
     * 检查并恢复未决事务
     */
    public void recover() {
        // 1. 从存储中加载所有活跃会话
        Collection<GlobalSession> allSessions = sessionManager.allSessions();
        
        // 2. 处理每个会话
        for (GlobalSession session : allSessions) {
            processUndecidedSession(session);
        }
    }
    
    /**
     * 处理未决会话
     */
    private void processUndecidedSession(GlobalSession session) {
        switch (session.getStatus()) {
            case COMMITTING:
                // 之前正在提交
                // 继续提交
                doCommit(session);
                break;
                
            case ROLLBACKING:
                // 之前正在回滚
                // 继续回滚
                doRollback(session);
                break;
                
            case BEGIN:
                // 之前刚开启就崩溃了
                // 根据超时判断
                if (isTimeout(session)) {
                    doRollback(session);
                }
                break;
                
            default:
                // 其他状态，可能是异常情况
                handleException(session);
        }
    }
}
```

/**
 * 客户端重连
 */
public class NettyClient {
    
    private volatile boolean running = true;
    
    /**
     * 重连线程
     */
    public void reconnect() {
        while (running) {
            try {
                // 1. 检查连接状态
                if (!isConnected()) {
                    // 2. 尝试连接
                    connect();
                }
                
                // 3. 等待一段时间后再次检查
                Thread.sleep(RECONNECT_INTERVAL);
                
            } catch (Exception e) {
                log.error("Reconnect failed", e);
            }
        }
    }
    
    /**
     * 处理连接断开
     */
    public void handleDisconnect() {
        // 1. 标记连接断开
        connected.set(false);
        
        // 2. 停止心跳
        stopHeartbeat();
        
        // 3. 启动重连
        reconnect();
    }
}

## 选举机制

### Leader 选举

```java
/**
 * Leader 选举
 */

// Seata 使用 Raft 协议进行 Leader 选举
// 只有 Leader 节点处理写请求
// Follower 节点提供读服务

/**
 * 选举流程
 */
public class LeaderElection {
    
    /**
     * 发起选举
     */
    public void startElection() {
        // 1. 增加当前任期
        currentTerm++;
        
        // 2. 投票给自己
        votedFor = selfId;
        
        // 3. 向其他节点发送投票请求
        for (Node node : otherNodes) {
            sendRequestVote(node, new VoteRequest(currentTerm, selfId, lastLogIndex, lastLogTerm));
        }
        
        // 4. 等待投票结果
        int votes = 1;  // 已有 1 票（自己）
        for (VoteResponse response : collectVotes()) {
            if (response.term > currentTerm) {
                // 发现更大的任期，转为 Follower
                becomeFollower(response.term);
                return;
            }
            
            if (response.voteGranted) {
                votes++;
            }
        }
        
        // 5. 如果获得多数票，成为 Leader
        if (votes > (nodeCount / 2)) {
            becomeLeader();
        }
    }
}
```

## 总结

Seata 的高可用设计包括：

1. **Session 共享存储**：全局会话存储在共享存储中，集群间共享
2. **Lock 共享存储**：全局锁存储在共享存储中，保证锁一致性
3. **故障恢复**：重启后自动恢复未决事务
4. **客户端重连**：连接断开后自动重连
5. **Leader 选举**：通过 Raft 协议选举主节点

这些机制共同保证了 Seata Server 的高可用性。
