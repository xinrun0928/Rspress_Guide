# gRPC 核心概念与使用

HTTP/1.1 时代，我们用 RESTful API 做服务间通信。JSON 的可读性很好，但当 QPS 达到十万级别时，JSON 的序列化开销和 HTTP 头部的冗余就成了瓶颈。

gRPC 来了。它用 Protocol Buffers 做序列化，基于 HTTP/2 传输，支持双向流式调用。Netflix、Netflix 都在用，你还有什么理由不学？

## 为什么选择 gRPC？

先看看 gRPC 和 REST 的对比：

| 特性 | REST | gRPC |
|---|---|---|
| 传输协议 | HTTP/1.1 | HTTP/2 |
| 序列化 | JSON | Protocol Buffers |
| 数据格式 | 文本 | 二进制 |
| 传输效率 | 低 | 高（压缩后体积小 3-10 倍） |
| 代码生成 | OpenAPI/Swagger | protobuf 编译器 |
| 双向流 | 不支持 | 支持 |
| 浏览器支持 | 好 | 有限（需要 grpc-web） |
| 学习曲线 | 低 | 中 |

## Protocol Buffers：高效的序列化

### 定义服务

创建一个 `.proto` 文件：

```protobuf
// user.proto
syntax = "proto3";

package user;

// 定义服务
service UserService {
    // 简单 RPC
    rpc GetUser(GetUserRequest) returns (User);

    // 服务端流式 RPC
    rpc ListUsers(ListUsersRequest) returns (stream User);

    // 客户端流式 RPC
    rpc CreateUsers(stream CreateUserRequest) returns (CreateUsersResponse);

    // 双向流式 RPC
    rpc StreamChat(stream ChatMessage) returns (stream ChatMessage);
}

// 定义消息
message GetUserRequest {
    int64 id = 1;
}

message User {
    int64 id = 1;
    string name = 2;
    string email = 3;
    repeated string roles = 4;  // repeated = 数组
    map<string, string> attrs = 5;  // map 类型
}

message ListUsersRequest {
    int32 page_size = 1;
    int32 page_num = 2;
    string keyword = 3;
}

message CreateUserRequest {
    string name = 1;
    string email = 2;
}

message CreateUsersResponse {
    int32 success_count = 1;
    repeated int64 user_ids = 2;
}

message ChatMessage {
    string content = 1;
    int64 timestamp = 2;
}
```

### 字段编号的秘密

每个字段都有一个编号，这是 Protocol Buffers 的精髓：

```protobuf
message User {
    int64 id = 1;      // 编号 1
    string name = 2;   // 编号 2
    string email = 3;  // 编号 3
}
```

编号的作用是**二进制序列化时的字段标识**。发送时，只发「编号 + 值」，不发明字段名。这就是为什么 PB 比 JSON 小这么快的原因。

> [!WARNING]
> 编号 **1-15** 用 1 个字节编码，**16-2047** 用 2 个字节。所以高频字段应该用 1-15 的编号。

### 生成代码

```bash
# 安装 protoc 编译器
brew install protobuf

# 安装 gRPC 插件
go install google.golang.org/protobuf/cmd/protoc-gen-go@latest
go install google.golang.org/grpc/cmd/protoc-gen-go-grpc@latest

# 生成 Go 代码
protoc --go_out=. --go_opt=paths=source_relative \
       --go-grpc_out=. --go-grpc_opt=paths=source_relative \
       user.proto
```

生成的代码包含：

- 序列化/反序列化方法
- gRPC 服务端接口
- gRPC 客户端存根

## 服务端开发

### Go 服务端

```go
package main

import (
    "context"
    "log"
    "net"

    "google.golang.org/grpc"
    pb "example/user"
)

type userServer struct {
    pb.UnimplementedUserServiceServer
    users map[int64]*pb.User
}

func (s *userServer) GetUser(ctx context.Context, req *pb.GetUserRequest) (*pb.User, error) {
    user, ok := s.users[req.Id]
    if !ok {
        return nil, fmt.Errorf("user not found: %d", req.Id)
    }
    return user, nil
}

func (s *userServer) ListUsers(req *pb.ListUsersRequest, stream pb.UserService_ListUsersServer) error {
    for _, user := range s.users {
        if err := stream.Send(user); err != nil {
            return err
        }
    }
    return nil
}

func (s *userServer) CreateUsers(stream pb.UserService_CreateUsersServer) error {
    var createdIds []int64
    for {
        req, err := stream.Recv()
        if err == io.EOF {
            return stream.SendAndClose(&pb.CreateUsersResponse{
                SuccessCount: int32(len(createdIds)),
                UserIds:      createdIds,
            })
        }
        if err != nil {
            return err
        }
        // 创建用户逻辑
        userId := int64(len(s.users) + 1)
        s.users[userId] = &pb.User{
            Id:    userId,
            Name:  req.Name,
            Email: req.Email,
        }
        createdIds = append(createdIds, userId)
    }
}

func (s *userServer) StreamChat(stream pb.UserService_StreamChatServer) error {
    for {
        msg, err := stream.Recv()
        if err == io.EOF {
            return nil
        }
        if err != nil {
            return err
        }
        // 处理消息并回复
        reply := &pb.ChatMessage{
            Content:    "Received: " + msg.Content,
            Timestamp:  time.Now().Unix(),
        }
        if err := stream.Send(reply); err != nil {
            return err
        }
    }
}

func main() {
    lis, _ := net.Listen("tcp", ":50051")
    s := grpc.NewServer()
    pb.RegisterUserServiceServer(s, &userServer{
        users: make(map[int64]*pb.User),
    })
    log.Fatal(s.Serve(lis))
}
```

### Java 服务端

```java
public class UserServiceImpl extends UserServiceGrpc.UserServiceImplBase {

    private final Map<Long, User> users = new ConcurrentHashMap<>();

    @Override
    public void getUser(GetUserRequest request, StreamObserver<User> responseObserver) {
        User user = users.get(request.getId());
        if (user == null) {
            responseObserver.onError(Status.NOT_FOUND
                .withDescription("User not found")
                .asRuntimeException());
            return;
        }
        responseObserver.onNext(user);
        responseObserver.onCompleted();
    }

    @Override
    public void listUsers(ListUsersRequest request, StreamObserver<User> responseObserver) {
        users.values().forEach(responseObserver::onNext);
        responseObserver.onCompleted();
    }

    @Override
    public StreamObserver<CreateUserRequest> createUsers(
            StreamObserver<CreateUsersResponse> responseObserver) {
        return new StreamObserver<>() {
            private final List<Long> createdIds = new ArrayList<>();

            @Override
            public void onNext(CreateUserRequest request) {
                long id = users.size() + 1;
                User user = User.newBuilder()
                    .setId(id)
                    .setName(request.getName())
                    .setEmail(request.getEmail())
                    .build();
                users.put(id, user);
                createdIds.add(id);
            }

            @Override
            public void onError(Throwable t) {
                log.error("Create users error", t);
            }

            @Override
            public void onCompleted() {
                responseObserver.onNext(CreateUsersResponse.newBuilder()
                    .setSuccessCount(createdIds.size())
                    .addAllUserIds(createdIds)
                    .build());
                responseObserver.onCompleted();
            }
        };
    }
}

// 启动服务
public class GrpcServer {
    public static void main(String[] args) throws IOException {
        Server server = ServerBuilder.forPort(50051)
            .addService(new UserServiceImpl())
            .build()
            .start();

        Runtime.getRuntime().addShutdownHook(new Thread(server::shutdown));
        server.awaitTermination();
    }
}
```

## 客户端开发

### Go 客户端

```go
package main

import (
    "context"
    "log"
    "time"

    "google.golang.org/grpc"
    "google.golang.org/grpc/credentials/insecure"
    pb "example/user"
)

func main() {
    // 1. 建立连接
    conn, err := grpc.Dial("localhost:50051",
        grpc.WithTransportCredentials(insecure.NewCredentials()))
    if err != nil {
        log.Fatalf("did not connect: %v", err)
    }
    defer conn.Close()

    // 2. 创建客户端
    client := pb.NewUserServiceClient(conn)

    // 3. 简单 RPC 调用
    ctx, cancel := context.WithTimeout(context.Background(), time.Second)
    defer cancel()

    user, err := client.GetUser(ctx, &pb.GetUserRequest{Id: 1})
    if err != nil {
        log.Fatalf("get user failed: %v", err)
    }
    log.Printf("User: %s", user.Name)

    // 4. 服务端流式调用
    stream, err := client.ListUsers(ctx, &pb.ListUsersRequest{PageSize: 10})
    for {
        user, err := stream.Recv()
        if err == io.EOF {
            break
        }
        log.Printf("User from stream: %s", user.Name)
    }

    // 5. 双向流式调用
    streamChat, err := client.StreamChat(ctx)
    go func() {
        for i := 0; i < 5; i++ {
            streamChat.Send(&pb.ChatMessage{
                Content:   fmt.Sprintf("Hello %d", i),
                Timestamp: time.Now().Unix(),
            })
            time.Sleep(time.Second)
        }
        streamChat.CloseSend()
    }()
    for {
        msg, err := streamChat.Recv()
        if err == io.EOF {
            break
        }
        log.Printf("Received: %s", msg.Content)
    }
}
```

### 客户端连接池

```go
type GrpcPool struct {
    pool sync.Pool
}

func NewGrpcPool(target string, size int) (*GrpcPool, error) {
    pool := &GrpcPool{
        pool: sync.Pool{
            New: func() interface{} {
                conn, err := grpc.Dial(target,
                    grpc.WithTransportCredentials(insecure.NewCredentials()))
                if err != nil {
                    return nil
                }
                return conn
            },
        },
    }
    return pool, nil
}

func (p *GrpcPool) Get() *grpc.ClientConn {
    return p.pool.Get().(*grpc.ClientConn)
}

func (p *GrpcPool) Put(conn *grpc.ClientConn) {
    p.pool.Put(conn)
}
```

## 拦截器：日志、认证、限流

gRPC 的拦截器分两种：**UnaryInterceptor**（普通调用）和 **StreamInterceptor**（流式调用）：

```go
// 服务端拦截器
func unaryInterceptor(ctx context.Context, req interface{},
    info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (interface{}, error) {

    start := time.Now()

    // 前置处理
    log.Printf("Received: %s.%s", info.FullMethod, req)

    // 调用实际方法
    resp, err := handler(ctx, req)

    // 后置处理
    log.Printf("Completed: %s in %v", info.FullMethod, time.Since(start))

    return resp, err
}

// 注册拦截器
server := grpc.NewServer(
    grpc.UnaryInterceptor(unaryInterceptor),
    grpc.StreamInterceptor(streamInterceptor),
)
```

```go
// 客户端拦截器
func unaryClientInterceptor(ctx context.Context, method string, req, reply interface{},
    cc *grpc.ClientConn, invoker grpc.UnaryInvoker, opts ...grpc.CallOption) error {

    // 添加认证信息
    md, ok := metadata.FromOutgoingContext(ctx)
    if !ok {
        md = metadata.New()
    }
    md.Set("authorization", "Bearer "+getToken())

    ctx = metadata.NewOutgoingContext(ctx, md)
    return invoker(ctx, method, req, reply, cc, opts...)
}

conn, _ := grpc.Dial("localhost:50051",
    grpc.WithUnaryInterceptor(unaryClientInterceptor))
```

## 元数据与认证

### 元数据传递

```go
// 服务端读取元数据
func (s *server) GetUser(ctx context.Context, req *pb.GetUserRequest) (*pb.User, error) {
    md, ok := metadata.FromIncomingContext(ctx)
    if ok {
        if token := md.Get("authorization"); len(token) > 0 {
            log.Printf("Token: %s", token[0])
        }
    }
    return &pb.User{Id: 1, Name: "John"}, nil
}

// 客户端添加元数据
func withAuth(ctx context.Context, token string) context.Context {
    md := metadata.Pairs("authorization", "Bearer "+token)
    return metadata.NewOutgoingContext(ctx, md)
}

// 使用
ctx := withAuth(context.Background(), "xxx-token")
user, _ := client.GetUser(ctx, &pb.GetUserRequest{Id: 1})
```

### Token 认证

```go
// 自定义认证
type Auth struct {
    token string
}

func (a *Auth) GetRequestMetadata(ctx context.Context, uri ...string) (map[string]string, error) {
    return map[string]string{"authorization": a.token}, nil
}

func (a *Auth) RequireTransportSecurity() bool {
    return false  // 生产环境应该返回 true
}

// 客户端使用
creds := grpc.WithPerRPCCredentials(&Auth{token: "your-token"})
conn, _ := grpc.Dial("localhost:50051", creds)
```

## 总结

gRPC 的核心能力，可以用这张图概括：

```
┌─────────────────────────────────────────────────────────────┐
│                        gRPC                                 │
├─────────────────────────────────────────────────────────────┤
│  ┌─────────────┐    ┌─────────────┐    ┌─────────────┐   │
│  │  Proto file │ -> │ Code Gen    │ -> │  Client/Server│   │
│  └─────────────┘    └─────────────┘    └─────────────┘   │
│                            │                               │
│                     HTTP/2 + Protobuf                      │
├─────────────────────────────────────────────────────────────┤
│  四种调用模式：                                             │
│  1. Simple RPC      - 一次请求，一次响应                   │
│  2. Server Streaming - 一次请求，服务器流响应              │
│  3. Client Streaming - 客户端流请求，服务器一次响应        │
│  4. Bidirectional   - 双向流                              │
└─────────────────────────────────────────────────────────────┘
```

选择 gRPC 的场景：

- 微服务内部通信，需要高性能、低延迟
- 多语言环境，Proto 生成各语言代码
- 需要双向流式通信
- 对带宽敏感，需要高效序列化

不适合的场景：

- 需要浏览器直接访问（用 REST 或 grpc-web）
- 对可读性要求高（调试时 JSON 更方便）
- 团队不熟悉 Protocol Buffers
